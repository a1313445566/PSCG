#### 触发条件

**文件**: `routes/answer-behavior.js`

```javascript
/**
 * POST /api/answer-behavior/analyze-style
 * 触发学习风格分析
 */
router.post('/analyze-style', async (req, res) => {
  const { userId } = req.body;
  
  // 检查答题数据是否足够
  const behaviorCount = await db.get(
    'SELECT COUNT(*) as count FROM answer_behavior WHERE user_id = ?',
    [userId]
  );
  
  if (behaviorCount.count < 10) {
    return res.status(400).json({ 
      error: '答题数据不足，至少需要 10 次答题记录'
    });
  }
  
  // 添加到任务队列，自动执行 AI 分析
  const taskId = await queueService.addTask('user_learning_style', userId, 5);
  
  res.json({ success: true, taskId });
});
```

**触发方式**：
1. **手动触发**：调用 `/api/answer-behavior/analyze-style`
2. **自动触发**：当查询用户学习风格且数据过期时（超过7天）

**前提条件**：
- ⚠️ `answer_behavior` 表至少有 10 条记录

### 方案一：集成答题行为追踪（推荐）

**步骤**：

1. **修改 QuizView.vue**

```vue
<template>
  <div class="quiz-view">
    <!-- 添加答题行为追踪组件 -->
    <AnswerBehaviorTracker ref="behaviorTracker" />
    
    <!-- 其他内容 -->
    <QuestionCard 
      v-for="question in currentQuestions"
      :key="question.id"
      :question="question"
      @select-option="(option) => {
        selectOption(question.id, option, question.type);
        // 追踪答案修改
        behaviorTracker?.trackModification();
        // 追踪首次答案
        behaviorTracker?.trackFirstAnswer(option);

    />
  </div>
</template>

<script setup>

// 获取组件引用

// 当开始答题时

  behaviorTracker.value?.startTracking(questionId)
}

  // 提交前先提交行为数据
```

2. **在 QuestionCard.vue 中添加追踪**

```vue
<script setup>
const props = defineProps({
  question: Object,
  showResult: Boolean
})

const emit = defineEmits(['select-option'])

// 记录鼠标悬停（犹豫）
const handleMouseEnter = () => {
  if (!props.showResult) {
    emit('hover-start')
  }
}

const handleMouseLeave = () => {
  if (!props.showResult) {
    emit('hover-end')
  }
}

const handleSelectOption = (option) => {
  if (!props.showResult) {
    emit('select-option', option)
  }
}
</script>

<template>
  <div 
    class="question-card"
    @mouseenter="handleMouseEnter"
    @mouseleave="handleMouseLeave"
  >
    <!-- 题目内容 -->
  </div>
</template>
```

```javascript
const db = require('../services/database');

async function generateMockBehaviorData() {

      `, [

    console.log(`✅ 为学生 ${student.id} 生成了 20 条行为数据`);
```
