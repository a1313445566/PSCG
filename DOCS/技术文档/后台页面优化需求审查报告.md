## 三、API封装问题

### 1. P1-1：API调用封装
**问题**：
- 缺少请求取消机制
- 防抖逻辑可能导致需要立即执行的请求被延迟
- 错误处理不够详细
- 缺少请求超时设置

**建议**：
```javascript
class ApiClient {
  constructor() {
    this.baseUrl = getApiBaseUrl()
    this.pending = new Map()
    this.abortControllers = new Map()
  }
  
  async request(endpoint, options = {}, config = {}) {
    const { retries = 2, showError = true, debounce = true, timeout = 30000 } = config
    
    // 请求超时处理
    const controller = new AbortController()
    const timeoutId = setTimeout(() => controller.abort(), timeout)
    this.abortControllers.set(endpoint, controller)
    
    try {
      const result = await this._doRequest(endpoint, { ...options, signal: controller.signal }, retries, showError)
      clearTimeout(timeoutId)
      return result
    } catch (error) {
      clearTimeout(timeoutId)
      if (error.name === 'AbortError') {
        throw new Error('请求超时')
      }
      throw error
    } finally {
      this.abortControllers.delete(endpoint)
    }
  }
  
  cancelRequest(endpoint) {
    const controller = this.abortControllers.get(endpoint)
    if (controller) {
      controller.abort()
      this.abortControllers.delete(endpoint)
    }
  }
  
  cancelAllRequests() {
    this.abortControllers.forEach(controller => controller.abort())
    this.abortControllers.clear()
  }
}
```

- 缺少跳转到指定页的功能

**建议**：

// src/composables/usePagination.js
export function usePagination(defaultPageSize = 10, total = ref(0)) {
  const currentPage = ref(1)
  const pageSize = ref(defaultPageSize)
  
  // 计算总页数
  const totalPages = computed(() => Math.ceil(total.value / pageSize.value))
  
  // 前端分页
  const paginate = (data) => {
    const start = (currentPage.value - 1) * pageSize.value
    return data.slice(start, start + pageSize.value)
  }
  
  // 服务端分页参数
  const getServerParams = () => ({
    page: currentPage.value,
    limit: pageSize.value
  })
  
  const handleSizeChange = (size) => { 
    pageSize.value = size
    currentPage.value = 1
  }
  
  const handleCurrentChange = (page) => { 
    currentPage.value = page 
  }
  
  const jumpToPage = (page) => {
    if (page >= 1 && page <= totalPages.value) {
      currentPage.value = page
    }
  }
  
  return { 
    currentPage, 
    pageSize, 
    totalPages,
    paginate, 
    getServerParams,
    handleSizeChange, 
    handleCurrentChange,
    jumpToPage
  }
}
```

**建议**：

// src/composables/useUserFilters.js
export function useUserFilters() {
  const studentId = ref('')
  const grade = ref('')
  const classNum = ref('')
  const subject = ref('')
  
  // 验证筛选条件
  const validate = () => {
    const errors = []
    if (studentId.value && !/^\d*$/.test(studentId.value)) {
      errors.push('学号只能包含数字')
    }
    return errors
  }
  
  const reset = () => { 
    studentId.value = ''; 
    grade.value = ''; 
    classNum.value = ''; 
    subject.value = '' 
  }
  
  const toParams = () => {
    const params = {}
    if (studentId.value) params.student_id = studentId.value
    if (grade.value) params.grade = grade.value
    if (classNum.value) params.class = classNum.value
    if (subject.value) params.subjectId = subject.value
    return params
  }
  
  const hasActiveFilters = computed(() => {
    return !!(studentId.value || grade.value || classNum.value || subject.value)
  })
  
  return { 
    studentId, 
    grade, 
    classNum, 
    subject, 
    validate,
    reset, 
    toParams,
    hasActiveFilters
  }
}
```
