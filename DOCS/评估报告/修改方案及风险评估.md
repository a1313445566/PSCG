### 2.2 SQL 注入漏洞修复

#### 问题描述
- **位置**：[routes/users.js:15-62](file:///e:/PSCG/routes/users.js#L15-L62)
- **风险**：攻击者可通过 SQL 注入获取/篡改数据
- **严重程度**：高危

#### 修改方案

**修改前（存在注入风险）：**

```javascript
// 危险：直接拼接 SQL
const query = `SELECT * FROM users WHERE username = '${username}'`
```

**修改后（参数化查询）：**

```javascript
// 安全：使用参数化查询
const query = 'SELECT * FROM users WHERE username = ?'
const [rows] = await pool.execute(query, [username])
```

**完整修改示例：**

```javascript
// routes/users.js 用户列表接口修改
router.get('/users', adminAuth, async (req, res) => {
  try {
    const { page = 1, pageSize = 10, keyword = '' } = req.query
    
    // 参数化查询
    let query = 'SELECT id, username, email, role, created_at FROM users WHERE 1=1'
    const params = []
    
    if (keyword) {
      query += ' AND (username LIKE ? OR email LIKE ?)'
      params.push(`%${keyword}%`, `%${keyword}%`)
    }
    
    query += ' ORDER BY created_at DESC LIMIT ? OFFSET ?'
    params.push(parseInt(pageSize), (parseInt(page) - 1) * parseInt(pageSize))
    
    const [rows] = await pool.execute(query, params)
    
    res.json({ code: 200, msg: '获取成功', data: rows })
  } catch (error) {
    console.error('获取用户列表失败:', error)
    res.status(500).json({ code: 500, msg: '服务器错误', data: null })
  }
})
```

#### 风险评估

| 风险项 | 风险等级 | 缓解措施 |
|--------|---------|----------|
| 参数类型转换错误 | 低 | 使用 parseInt 明确转换 |
| 查询性能变化 | 极低 | 参数化查询性能更优 |
| 现有功能回归 | 中 | 完整回归测试 |

```javascript
// routes/upload.js 增强文件验证
const FILE_TYPE_CHECKS = {
  image: {
    mimeTypes: ['image/jpeg', 'image/png', 'image/gif', 'image/webp'],
    maxSise: 5 * 1024 * 1024, // 5MB
    magicNumbers: {
      'image/jpeg': [0xFF, 0xD8, 0xFF],
      'image/png': [0x89, 0x50, 0x4E, 0x47],
      'image/gif': [0x47, 0x49, 0x46, 0x38]
    }
  },
  audio: {
    mimeTypes: ['audio/mpeg', 'audio/wav', 'audio/ogg'],
    maxSize: 20 * 1024 * 1024 // 20MB
  }
}

// 文件内容验证函数
const validateFileContent = (buffer, expectedType) => {
  const checks = FILE_TYPE_CHECKS[expectedType]
  if (!checks) return false
  
  // 验证魔数（文件头）
  for (const [mimeType, magicNumbers] of Object.entries(checks.magicNumbers || {})) {
    const header = buffer.slice(0, magicNumbers.length)
    if (magicNumbers.every((byte, i) => header[i] === byte)) {
      return true
    }
  }
  return false
}
```

```javascript
// utils/xss-filter.js 增强 XSS 过滤规则
const XSS_FILTER_RULES = {
  // 基础过滤
  basic: {
    '<script': '&lt;script',
    '</script': '&lt;/script',
    'javascript:': 'java_script:',
    'onerror': 'on_error',
    'onload': 'on_load',
    'onclick': 'on_click'
  },
  // 高级过滤
  advanced: {
    '<iframe': '&lt;iframe',
    '<object': '&lt;object',
    '<embed': '&lt;embed',
    'expression(': 'expression_(',
    'vbscript:': 'vb_script:',
    'data:text/html': 'data:text/plain'
  }
}

// 增强过滤函数
const sanitizeHTML = (input) => {
  if (typeof input !== 'string') return input
  
  let result = input
  
  // 应用所有过滤规则
  Object.values(XSS_FILTER_RULES).forEach(rules => {
    Object.entries(rules).forEach(([pattern, replacement]) => {
      const regex = new RegExp(pattern, 'gi')
      result = result.replace(regex, replacement)
    })
  })
  
  return result
}
```

### 4.3 虚拟滚动实现

#### 组件实现

```vue
<!-- src/components/common/VirtualList.vue -->
<script setup>
import { ref, computed, onMounted, onUnmounted } from 'vue'

const props = defineProps({
  items: { type: Array, required: true },
  itemHeight: { type: Number, default: 50 },
  visibleCount: { type: Number, default: 20 }
})

const containerRef = ref(null)
const scrollTop = ref(0)

const startIndex = computed(() => Math.floor(scrollTop.value / props.itemHeight))
const endIndex = computed(() => Math.min(startIndex.value + props.visibleCount, props.items.length))
const visibleItems = computed(() => props.items.slice(startIndex.value, endIndex.value))
const totalHeight = computed(() => props.items.length * props.itemHeight)
const offsetY = computed(() => startIndex.value * props.itemHeight)

const handleScroll = (e) => {
  scrollTop.value = e.target.scrollTop
}

onMounted(() => {
  containerRef.value?.addEventListener('scroll', handleScroll)
})

onUnmounted(() => {
  containerRef.value?.removeEventListener('scroll', handleScroll)
})
</script>

<template>
  <div ref="containerRef" class="virtual-list-container">
    <div class="virtual-list-phantom" :style="{ height: `${totalHeight}px` }"></div>
    <div class="virtual-list-content" :style="{ transform: `translateY(${offsetY}px)` }">
      <div v-for="(item, index) in visibleItems" :key="startIndex + index" class="virtual-list-item">
        <slot :item="item" :index="startIndex + index"></slot>
      </div>
    </div>
  </div>
</template>

<style scoped>
.virtual-list-container {
  height: 100%;
  overflow-y: auto;
  position: relative;
}

.virtual-list-phantom {
  position: absolute;
  left: 0;
  top: 0;
  right: 0;
  z-index: -1;
}

.virtual-list-content {
  position: absolute;
  left: 0;
  right: 0;
  top: 0;
}
</style>
```

```javascript
  test('应拒绝包含 SQL 注入的用户名', async () => {
    const maliciousInputs = [
      "admin'--",
      "admin' OR '1'='1",
      "admin'; DROP TABLE users--"
    ]
    
    for (const input of maliciousInputs) {
      const res = await request(app)
        .get('/api/users')
        .query({ keyword: input })
      
      expect(res.status).toBe(200)
      expect(res.body.data).not.toContainEqual(
        expect.objectContaining({ role: 'admin' })
      )
    }
  })
})
```
