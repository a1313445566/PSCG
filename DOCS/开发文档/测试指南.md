# 🧪 PSCG 测试指南

本文档详细介绍了 PSCG 项目的测试策略、测试类型和测试执行流程。

## 1. 测试策略

### 1.1 测试目标
- 确保系统功能正确性
- 验证系统性能指标
- 保障代码质量
- 减少回归错误
- 提高开发效率

### 1.2 测试层级
```
┌─────────────────────────────────────────┐
│           E2E 测试（端到端）            │
├─────────────────────────────────────────┤
│        集成测试（组件/模块）             │
├─────────────────────────────────────────┤
│          单元测试（函数/类）            │
└─────────────────────────────────────────┘
```

### 1.3 测试工具栈
| 测试类型 | 工具 | 用途 |
|----------|------|------|
| 单元测试 | Jest | JavaScript 测试 |
| 组件测试 | Vue Test Utils | Vue 组件测试 |
| E2E 测试 | Cypress | 端到端测试 |
| API 测试 | Supertest | API 接口测试 |
| 性能测试 | K6 | 性能负载测试 |
| 安全测试 | OWASP ZAP | 安全漏洞扫描 |

## 3. 集成测试

### 3.1 API 集成测试
```javascript
// tests/integration/api/users.spec.js
const request = require('supertest')
const app = require('../../server.cjs')

describe('用户 API', () => {
  let token

  beforeAll(async () => {
    // 获取认证令牌
    const res = await request(app)
      .post('/api/auth/login')
      .send({
        username: 'admin',
        password: 'admin123'
      })
    
    token = res.body.token
  })

  describe('GET /api/users', () => {
    it('应该返回用户列表', async () => {
      const res = await request(app)
        .get('/api/users')
        .set('Authorization', `Bearer ${token}`)
        .expect(200)

      expect(res.body.success).toBe(true)
      expect(Array.isArray(res.body.data)).toBe(true)
    })

    it('未认证应该返回401', async () => {
      await request(app)
        .get('/api/users')
        .expect(401)
    })
  })

  describe('POST /api/users', () => {
    it('应该创建新用户', async () => {
      const newUser = {
        username: 'testuser',
        email: 'test@example.com',
        password: 'Test123!',
        role: 'student'
      }

      const res = await request(app)
        .post('/api/users')
        .set('Authorization', `Bearer ${token}`)
        .send(newUser)
        .expect(201)

      expect(res.body.success).toBe(true)
      expect(res.body.data.username).toBe('testuser')
    })
  })
})
```

### 3.2 数据库集成测试
```javascript
// tests/integration/database/userService.spec.js
const db = require('../../services/database')
const UserService = require('../../services/userService')

describe('UserService 集成测试', () => {
  beforeAll(async () => {
    // 建立数据库连接
    await db.connect()
  })

  afterAll(async () => {
    // 清理测试数据
    await db.query('DELETE FROM users WHERE email LIKE ?', ['test%@example.com'])
    await db.disconnect()
  })

  beforeEach(async () => {
    // 每个测试前重置数据
    await db.query('DELETE FROM users WHERE email = ?', ['integration-test@example.com'])
  })

  it('应该创建和检索用户', async () => {
    const userData = {
      username: 'integrationuser',
      email: 'integration-test@example.com',
      password: 'Test123!'
    }

    // 创建用户
    const createdUser = await UserService.createUser(userData)
    expect(createdUser.id).toBeDefined()
    expect(createdUser.email).toBe(userData.email)

    // 检索用户
    const retrievedUser = await UserService.getUserById(createdUser.id)
    expect(retrievedUser.email).toBe(userData.email)
  })
})
```

```javascript
// tests/e2e/login.cy.js
describe('登录流程', () => {
  beforeEach(() => {
    // 每次测试前访问登录页
    cy.visit('/login')
  })

  it('应该成功登录', () => {
    // 输入用户名和密码
    cy.get('[data-test="username-input"]').type('student1')
    cy.get('[data-test="password-input"]').type('password123')
    
    // 点击登录按钮
    cy.get('[data-test="login-button"]').click()
    
    // 验证重定向到首页
    cy.url().should('include', '/dashboard')
    
    // 验证用户信息显示
    cy.get('[data-test="user-name"]').should('contain', '学生1')
  })

  it('错误的密码应该显示错误信息', () => {
    cy.get('[data-test="username-input"]').type('student1')
    cy.get('[data-test="password-input"]').type('wrongpassword')
    cy.get('[data-test="login-button"]').click()
    
    // 验证错误信息
    cy.get('[data-test="error-message"]')
      .should('be.visible')
      .and('contain', '用户名或密码错误')
  })
})
```

## 6. 性能测试

### 6.1 K6 负载测试
```javascript
// tests/performance/login-load-test.js
import http from 'k6/http'
import { check, sleep } from 'k6'
import { Rate } from 'k6/metrics'

// 自定义指标
const errorRate = new Rate('errors')

// 测试配置
export const options = {
  stages: [
    { duration: '30s', target: 20 },    // 逐步增加到20个用户
    { duration: '1m', target: 20 },     // 保持20个用户1分钟
    { duration: '30s', target: 0 },     // 逐步减少到0
  ],
  thresholds: {
    http_req_duration: ['p(95)<500'],   // 95%的请求应在500ms内完成
    errors: ['rate<0.1'],               // 错误率低于10%
  }
}

export default function () {
  const url = 'http://localhost:3001/api/auth/login'
  const payload = JSON.stringify({
    username: 'testuser',
    password: 'testpass'
  })

  const params = {
    headers: {
      'Content-Type': 'application/json',
    },
  }

  const res = http.post(url, payload, params)

  // 检查响应
  const success = check(res, {
    '状态码是200': (r) => r.status === 200,
    '响应包含token': (r) => r.json().token !== undefined,
  })

  // 记录错误
  errorRate.add(!success)

  // 模拟用户思考时间
  sleep(1)
}
```

```yaml
# .github/workflows/test.yml
name: 测试流程

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

jobs:
  test:
    runs-on: ubuntu-latest
    
    strategy:
      matrix:
        node-version: [16.x, 18.x]
    
    steps:
    - uses: actions/checkout@v3

    - name: 安装依赖

    - name: 运行集成测试
      run: npm run test:integration
    
    - name: 运行 API 测试
      run: npm run test:api
    
    - name: 上传测试报告
      uses: actions/upload-artifact@v3
      if: always()
      with:
        name: 测试报告
```

### 9.2 测试结构
```javascript
// AAA 模式：Arrange, Act, Assert
describe('Calculator', () => {
  it('应该正确相加两个数字', () => {
    // Arrange: 准备测试数据
    const calculator = new Calculator()
    const a = 5
    const b = 3
    
    // Act: 执行被测试的操作
    const result = calculator.add(a, b)
    
    // Assert: 验证结果
    expect(result).toBe(8)
  })
})
```

### 9.3 测试隔离
```javascript
// 使用 beforeEach 和 afterEach 保持测试独立
describe('UserService', () => {
  let userService
  let mockDb
  
  beforeEach(() => {
    // 每个测试前创建新实例
    mockDb = {
      query: jest.fn()
    }
    userService = new UserService(mockDb)
  })
  
  afterEach(() => {
    // 清理模拟
    jest.clearAllMocks()
  })
})
```
