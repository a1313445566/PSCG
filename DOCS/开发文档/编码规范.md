### 2. 文件命名规范

#### Vue组件文件
- 使用**PascalCase**（大驼峰命名法）
- 组件名与文件名一致
- 使用`.vue`扩展名
- **正确示例**：`UserManagement.vue`, `QuestionCard.vue`
- **错误示例**：`userManagement.vue`, `question-card.vue`

#### JavaScript文件
- 使用**camelCase**（小驼峰命名法）
- 根据用途命名
- **工具函数**：`dateUtils.js`, `validationUtils.js`
- **服务文件**：`database.js`, `cacheService.js`
- **路由文件**：`users.js`, `questions.js`

#### CSS文件
- 使用**kebab-case**（短横线命名法）
- 描述性命名
- **示例**：`global.css`, `loading-animation.css`


#### Hook 编写规范

##### 基础结构
```javascript
import { ref, computed, onUnmounted } from 'vue'

/**
 * Hook 描述
 * @param {Object} options - 配置选项
 * @returns {Object} hook 方法与状态
 *
 * @example
 * const { data, isLoading, execute } = useXxx()
 * await execute()
 */
export function useXxx(options = {}) {
  // 1. 响应式状态
  const data = ref(null)
  const isLoading = ref(false)
  const error = ref(null)

  // 2. 内部资源（需要清理的）
  let timer = null
  let controller = null

  // 3. 核心方法
  const execute = async (params) => {
    isLoading.value = true
    error.value = null

    try {
      const result = await api.get('/endpoint', params)
      data.value = result
      return result
    } catch (err) {
      error.value = err.message
      console.error('[useXxx] 执行失败:', err)
      throw err // 向上抛出让调用者处理
    } finally {
      isLoading.value = false
    }
  }

  // 4. 重置方法
  const reset = () => {
    data.value = null
    error.value = null
    isLoading.value = false
  }

  // 5. 资源清理
  const cleanup = () => {
    if (timer) clearTimeout(timer)
    if (controller) controller.abort()
  }

  // 6. 组件卸载时自动清理
  onUnmounted(cleanup)

  // 7. 返回值（按逻辑分组）
  return {
    // 状态
    data,
    isLoading,
    error,

    // 计算属性
    isEmpty: computed(() => !data.value),
    hasError: computed(() => !!error.value),

    // 方法
    execute,
    reset,
    cleanup
  }
}
```

##### 必须遵守的规则

| 规则 | 说明 |
|------|------|
| 异步操作 try-catch | 所有 async 方法必须有错误处理 |
| 资源清理 | 定时器、事件监听、网络请求必须清理 |
| onUnmounted 清理 | 组件卸载时自动清理资源 |
| 错误向上抛出 | 让调用者决定如何处理错误 |
| 日志前缀 | console.error('[useXxx] 错误:', error) |

##### 资源清理模板

```javascript
export function useTimer() {
  // 定时器
  let timer = null

  const start = (callback, delay) => {
    if (timer) clearTimeout(timer) // 避免重复
    timer = setTimeout(callback, delay)
  }

  const stop = () => {
    if (timer) {
      clearTimeout(timer)
      timer = null
    }
  }

  // 必须在卸载时清理
  onUnmounted(stop)

  return { start, stop }
}

export function useFetch() {
  // 网络请求取消
  let controller = null

  const fetch = async (url) => {
    if (controller) controller.abort() // 取消上一个请求

    controller = new AbortController()

    try {
      const response = await fetch(url, { signal: controller.signal })
      return await response.json()
    } catch (error) {
      if (error.name === 'AbortError') {
        console.log('[useFetch] 请求已取消')
        return null
      }
      throw error
    }
  }

  const cancel = () => {
    if (controller) {
      controller.abort()
      controller = null
    }
  }

  onUnmounted(cancel)

  return { fetch, cancel }
}

export function useEventListener(target, event, handler) {
  // 事件监听
  onMounted(() => {
    target.addEventListener(event, handler)
  })

  // 必须移除监听
  onUnmounted(() => {
    target.removeEventListener(event, handler)
  })
}
```

```javascript
function errorHandler(err, req, res, next) {
  console.error('错误详情:', err)

  if (err.name === 'ValidationError') {

    // 开发环境显示详细错误
    ...(process.env.NODE_ENV === 'development' && { detail: err.message })
  })
}
```

```javascript
// services/database.js
const mysql = require('mysql2/promise')

class DatabaseService {
  constructor() {
    this.pool = mysql.createPool({
      host: process.env.DB_HOST,
      user: process.env.DB_USER,
      password: process.env.DB_PASSWORD,
      database: process.env.DB_NAME,
      waitForConnections: true,
      connectionLimit: 10,
      queueLimit: 0
    })
  }
  
  // 1. 使用连接池
  async query(sql, params = []) {
    const connection = await this.pool.getConnection()
    try {
      const [rows] = await connection.execute(sql, params)
      return rows
    } finally {
      connection.release()
    }
  }
  
  // 2. 事务处理
  async transaction(callback) {
    const connection = await this.pool.getConnection()
    await connection.beginTransaction()
    
    try {
      const result = await callback(connection)
      await connection.commit()
      return result
    } catch (error) {
      await connection.rollback()
      throw error
    } finally {
      connection.release()
    }
  }
  
  // 3. 参数化查询防止SQL注入
  async getUserById(id) {
    const sql = 'SELECT * FROM users WHERE id = ?'
    const [rows] = await this.query(sql, [id])
    return rows[0]
  }
  
  // 4. 批量操作
  async batchInsert(table, data) {
    if (!data || data.length === 0) return []
    
    const keys = Object.keys(data[0])
    const values = data.map(item => 
      keys.map(key => item[key])
    )
    
    const sql = `INSERT INTO ${table} (${keys.join(', ')}) VALUES ?`
    const result = await this.query(sql, [values])
    return result
  }
}
```
