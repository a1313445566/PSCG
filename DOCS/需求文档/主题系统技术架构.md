## 一、架构设计原则

### 1.1 核心原则

1. **模块化**：主题独立，互不影响
2. **可扩展**：支持自定义主题开发
3. **高性能**：懒加载，缓存优化
4. **易维护**：清晰的文件结构，完善的文档

### 1.2 设计模式

采用**策略模式** + **工厂模式**：

```
ThemeLoader (工厂)
    ├── DefaultTheme (策略)
    ├── OceanTheme (策略)
    └── ForestTheme (策略)
```

### 3.3 主题Store设计

```javascript
import { defineStore } from 'pinia'
import { themeLoader } from '@/utils/themeLoader'

export const useThemeStore = defineStore('theme', {
  state: () => ({
    currentTheme: localStorage.getItem('theme') || 'default',
    themes: {},
    themeConfig: {},
    isLoading: false,
    error: null
  }),

  getters: {
    themeList: state => Object.values(state.themes),
    currentThemeConfig: state => state.themeConfig[state.currentTheme] || {}
  },

  actions: {
    async loadThemes() {
      try {
        this.isLoading = true
        const themes = await api.get('/themes')
        this.themes = themes
      } catch (error) {
        this.error = error.message
        console.error('加载主题列表失败:', error)
      } finally {
        this.isLoading = false
      }
    },

    async setTheme(themeName) {
      try {
        this.isLoading = true
        
        const success = await themeLoader.loadTheme(themeName)
        
        if (success) {
          this.currentTheme = themeName
          localStorage.setItem('theme', themeName)
          
          await api.post('/themes/set', { theme: themeName })
          
          return true
        }
        
        return false
      } catch (error) {
        this.error = error.message
        console.error('主题切换失败:', error)
        return false
      } finally {
        this.isLoading = false
      }
    },

    async updateThemeConfig(config) {
      try {
        await api.post('/themes/config', {
          theme: this.currentTheme,
          config
        })
        
        this.themeConfig[this.currentTheme] = {
          ...this.themeConfig[this.currentTheme],
          ...config
        }
        
        themeLoader.applyThemeConfig(this.themeConfig[this.currentTheme])
        
        return true
      } catch (error) {
        this.error = error.message
        console.error('主题配置更新失败:', error)
        return false
      }
    }
  }
})
```

### 3.4 主题加载器设计

```javascript
export class ThemeLoader {
  constructor() {
    this.currentTheme = null
    this.themesPath = '/themes'
    this.loadedStyles = new Set()
  }

  async loadTheme(themeName) {
    try {
      console.log(`[ThemeLoader] 开始加载主题: ${themeName}`)
      
      const config = await this.loadThemeConfig(themeName)
      
      await this.loadThemeStyles(themeName)
      
      await this.loadThemeComponents(themeName)
      
      this.applyThemeConfig(config)
      
      this.currentTheme = themeName
      
      console.log(`[ThemeLoader] 主题加载成功: ${themeName}`)
      return true
    } catch (error) {
      console.error('[ThemeLoader] 主题加载失败:', error)
      return false
    }
  }

  async loadThemeConfig(themeName) {
    try {
      const config = await import(`@/themes/${themeName}/theme.json`)
      return config.default || config
    } catch (error) {
      console.warn(`[ThemeLoader] 主题配置加载失败，使用默认配置: ${themeName}`)
      return this.getDefaultConfig()
    }
  }

  async loadThemeStyles(themeName) {
    try {
      const stylePath = `@/themes/${themeName}/styles/index.scss`
      
      if (this.loadedStyles.has(stylePath)) {
        console.log(`[ThemeLoader] 样式已加载: ${stylePath}`)
        return
      }
      
      await import(stylePath)
      this.loadedStyles.add(stylePath)
      
      console.log(`[ThemeLoader] 样式加载成功: ${stylePath}`)
    } catch (error) {
      console.warn(`[ThemeLoader] 样式加载失败，使用默认样式: ${themeName}`)
      await import('@/styles/scss/main.scss')
    }
  }

  async loadThemeComponents(themeName) {
    try {
      await import(`@/themes/${themeName}/components/index.js`)
      console.log(`[ThemeLoader] 组件加载成功: ${themeName}`)
    } catch (error) {
      console.warn(`[ThemeLoader] 组件加载失败，使用默认组件: ${themeName}`)
    }
  }

  applyThemeConfig(config) {
    const root = document.documentElement
    
    Object.entries(config).forEach(([key, value]) => {
      const cssVarName = this.toCssVarName(key)
      root.style.setProperty(cssVarName, value)
    })
    
    console.log('[ThemeLoader] 主题配置应用成功')
  }

  toCssVarName(key) {
    return `--theme-${key.replace(/([A-Z])/g, '-$1').toLowerCase()}`
  }

  getDefaultConfig() {
    return {
      primaryColor: '#ff6b6b',
      secondaryColor: '#4ecdc4',
      accentColor: '#ffd166',
      borderColor: '#ffd166',
      backgroundGradient: 'linear-gradient(135deg, #f8f9fa 0%, #e3f2fd 100%)',
      cardShadow: '0 8px 25px rgba(0, 0, 0, 0.15)',
      layout: 'default'
    }
  }
}

export const themeLoader = new ThemeLoader()
```

### 6.2 数据库设计

#### themes 表
```sql
CREATE TABLE themes (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(50) NOT NULL UNIQUE,
  display_name VARCHAR(100) NOT NULL,
  version VARCHAR(20) NOT NULL,
  author VARCHAR(100),
  description TEXT,
  screenshot VARCHAR(255),
  is_active BOOLEAN DEFAULT FALSE,
  config JSON,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

#### user_themes 表
```sql
CREATE TABLE user_themes (
  id INT AUTO_INCREMENT PRIMARY KEY,
  user_id INT NOT NULL,
  theme_name VARCHAR(50) NOT NULL,
  config JSON,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES users(id),
  UNIQUE KEY unique_user_theme (user_id, theme_name)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

## 七、性能优化方案

### 7.1 懒加载策略

```javascript
const lazyLoadTheme = (themeName) => {
  return () => import(`@/themes/${themeName}/index.js`)
    .then(module => {
      return module.default
    })
    .catch(() => {
      return import('@/themes/default/index.js')
    })
}
```

### 7.2 缓存策略

```javascript
const themeCache = {
  get(themeName) {
    const cached = localStorage.getItem(`theme_cache_${themeName}`)
    if (cached) {
      const { data, timestamp } = JSON.parse(cached)
      if (Date.now() - timestamp < 3600000) {
        return data
      }
    }
    return null
  },
  
  set(themeName, data) {
    localStorage.setItem(`theme_cache_${themeName}`, JSON.stringify({
      data,
      timestamp: Date.now()
    }))
  }
}
```

### 7.3 预加载策略

```javascript
const preloadThemes = ['default', 'ocean']

preloadThemes.forEach(themeName => {
  const link = document.createElement('link')
  link.rel = 'preload'
  link.href = `/themes/${themeName}/styles/index.css`
  link.as = 'style'
  document.head.appendChild(link)
})
```

## 八、安全方案

### 8.1 主题文件验证

```javascript
const validateThemeFile = (file) => {
  const allowedExtensions = ['.json', '.scss', '.vue', '.js', '.png', '.jpg']
  const maxSize = 5 * 1024 * 1024
  
  const ext = path.extname(file.name)
  if (!allowedExtensions.includes(ext)) {
    throw new Error('不支持的文件类型')
  }
  
  if (file.size > maxSize) {
    throw new Error('文件大小超过限制')
  }
  
  return true
}
```

### 8.2 XSS防护

```javascript
const sanitizeThemeConfig = (config) => {
  const sanitized = {}
  
  Object.entries(config).forEach(([key, value]) => {
    if (typeof value === 'string') {
      sanitized[key] = xss(value)
    } else {
      sanitized[key] = value
    }
  })
  
  return sanitized
}
```
