```javascript
export function useLoading() {

  const showLoading = (text = '加载中...') => {
    // 先关闭已有实例
    if (loadingInstance.value) {

    showLoading(text)
    try {
      return await fn()
    } finally {
      hideLoading()
    }
  }
  
  return { showLoading, hideLoading, withLoading }
}
```

#### 需求 P0-3：缓存策略优化

**问题**：核心数据缓存 24 小时，后台需实时数据

**前台影响评估**：
- ⚠️ **有影响**：`questionStore.js` 被前后台共享
- 前台使用页面：HomeView、QuizView、SubcategoryView、ResultView、App.vue
- **风险**：缓存策略变更可能影响前台数据加载

**当前代码**（questionStore.js:173）：
```javascript
localStorage.setItem('coreDataExpiry', Date.now() + 24 * 60 * 60 * 1000) // 硬编码
```

**优化方案**（前后台缓存键分离，确保前台不受影响）：
```javascript
// src/utils/cacheConfig.js

// 缓存版本号（用于强制刷新缓存）
export const CACHE_VERSION = 'v1'

// 缓存键常量管理（前后台分离）
export const CACHE_KEYS = {
  // 前台缓存键（保持不变）
  CORE_DATA: `coreData_${CACHE_VERSION}`,
  CORE_DATA_EXPIRY: `coreDataExpiry_${CACHE_VERSION}`,
  
  // 后台缓存键（独立，不影响前台）
  ADMIN_CORE_DATA: `adminCoreData_${CACHE_VERSION}`,
  ADMIN_CORE_DATA_EXPIRY: `adminCoreDataExpiry_${CACHE_VERSION}`
}

// 缓存时间配置（毫秒）
export const CACHE_TTL = {
  ADMIN: 5 * 60 * 1000,          // 后台：5分钟
  FRONTEND: 24 * 60 * 60 * 1000  // 前台：24小时（保持不变）
}

// 判断是否为后台页面（基于路由，更可靠）
export const isAdminPage = () => {
  // 方案1：基于 URL 路径
  return window.location.pathname.startsWith('/admin')
  
  // 方案2：基于 Vue Router（更可靠，需要在组件中使用）
  // import { useRoute } from 'vue-router'
  // const route = useRoute()
  // return route.path.startsWith('/admin')
}

// 获取当前页面对应的缓存键
export const getCacheKeys = () => {
  return isAdminPage() 
    ? { 
        data: CACHE_KEYS.ADMIN_CORE_DATA, 
        expiry: CACHE_KEYS.ADMIN_CORE_DATA_EXPIRY 
      }
    : { 
        data: CACHE_KEYS.CORE_DATA, 
        expiry: CACHE_KEYS.CORE_DATA_EXPIRY 
      }
}

// 获取当前页面对应的缓存时间
export const getCacheTTL = () => {
  return isAdminPage() ? CACHE_TTL.ADMIN : CACHE_TTL.FRONTEND
}

// 安全的缓存读取（带损坏检测）
export const getCacheSafely = (key) => {
  try {
    const data = localStorage.getItem(key)
    if (!data) return null
    return JSON.parse(data)
  } catch (error) {
    console.warn(`缓存数据损坏，已清除: ${key}`)
    localStorage.removeItem(key)
    return null
  }
}

// 安全的缓存写入
export const setCacheSafely = (key, data) => {
  try {
    localStorage.setItem(key, JSON.stringify(data))
    return true
  } catch (error) {
    console.warn(`缓存写入失败: ${key}`, error)
    // 缓存空间不足时，清理旧缓存
    clearOldCache()
    return false
  }
}

// 清理旧版本缓存
export const clearOldCache = () => {
  const keysToRemove = []
  for (let i = 0; i < localStorage.length; i++) {
    const key = localStorage.key(i)
    // 清理不带版本号的旧缓存
    if (key && (key.startsWith('coreData') || key.startsWith('adminCoreData')) && !key.includes(CACHE_VERSION)) {
      keysToRemove.push(key)
    }
  }
  keysToRemove.forEach(key => localStorage.removeItem(key))
}
```

**questionStore.js 修改示例**：
```javascript
import { getCacheKeys, getCacheTTL, getCacheSafely, setCacheSafely } from '@/utils/cacheConfig'

// 加载核心数据
async loadData() {
  const { data: dataKey, expiry: expiryKey } = getCacheKeys()
  const ttl = getCacheTTL()
  
  // 尝试从缓存读取
  const cached = getCacheSafely(dataKey)
  const expiry = localStorage.getItem(expiryKey)
  
  if (cached && expiry && Date.now() < parseInt(expiry)) {
    this.subjects = cached.subjects
    this.grades = cached.grades
    this.classes = cached.classes
    return
  }
  
  // 从服务器获取
  const response = await fetch(`${getApiBaseUrl()}/core-data`)
  const data = await response.json()
  
  this.subjects = data.subjects
  this.grades = data.grades
  this.classes = data.classes
  
  // 写入缓存
  setCacheSafely(dataKey, {
    subjects: this.subjects,
    grades: this.grades,
    classes: this.classes
  })
  localStorage.setItem(expiryKey, Date.now() + ttl)
}
```

#### 需求 P0-4：服务端分页

**问题**：用户管理使用前端分页，大数据量性能差

**注意**：`SQL_CALC_FOUND_ROWS` 在 MySQL 8.0.17+ 已弃用，使用 `COUNT(*)` 替代

**后端方案**：
```javascript
// routes/users.js
router.get('/', async (req, res) => {
  try {
    const { 
      page = 1, 
      limit = 20, 
      student_id, 
      name, 
      grade, 
      class: classNum,
      withStats = false 
    } = req.query
    
    const pageNum = parseInt(page) || 1
    const limitNum = parseInt(limit) || 20
    
    // 构建基础查询条件
    const conditions = []
    const params = []
    
    if (student_id) {
      conditions.push('student_id LIKE ?')
      params.push(`%${student_id}%`)
    }
    if (name) {
      conditions.push('name LIKE ?')
      params.push(`%${name}%`)
    }
    if (grade) {
      conditions.push('grade = ?')
      params.push(grade)
    }
    if (classNum) {
      conditions.push('class = ?')
      params.push(classNum)
    }
    
    const whereClause = conditions.length > 0 
      ? `WHERE ${conditions.join(' AND ')}` 
      : ''
    
    // 1. 先获取总数（不使用 SQL_CALC_FOUND_ROWS）
    const countSql = `SELECT COUNT(*) as total FROM users ${whereClause}`
    const [[{ total }]] = await pool.query(countSql, params)
    
    // 2. 再获取分页数据
    const offset = (pageNum - 1) * limitNum
    const dataSql = `
      SELECT * FROM users ${whereClause}
      ORDER BY CAST(student_id AS UNSIGNED)
      LIMIT ? OFFSET ?
    `
    const [users] = await pool.query(dataSql, [...params, limitNum, offset])
    
    // 3. 可选：批量获取统计数据
    if (withStats === 'true' && users.length > 0) {
      // ... 批量查询统计信息
    }
    
    res.json({ 
      data: users, 
      total, 
      page: pageNum, 
      limit: limitNum,
      totalPages: Math.ceil(total / limitNum)
    })
  } catch (error) {
    console.error('获取用户列表失败:', error)
    res.status(500).json({ error: '获取用户列表失败' })
  }
})
```

#### 需求 P1-1：API 调用封装 ✅ 已完成

**问题**：50+ 处重复的 `try { fetch } catch { ElMessage.error }` 模式

**状态**：✅ 已实施（2026-03-25）| 文件：`src/utils/api.js`

**方案**：
```javascript
// src/utils/api.js
import { ElMessage } from 'element-plus'
import { getApiBaseUrl } from '@/config/api'
import { getCSRFToken } from '@/utils/csrf'

class ApiClient {
  constructor() {
    this.baseUrl = getApiBaseUrl()
    this.pending = new Map()
    this.abortControllers = new Map()
  }
  
  // 生成请求唯一标识
  _generateKey(endpoint, options) {
    return `${options.method || 'GET'}:${endpoint}:${JSON.stringify(options.body || {})}`
  }
  
  // 延迟函数
  _delay(ms) {
    return new Promise(resolve => setTimeout(resolve, ms))
  }
  
  // 判断是否为网络错误
  _isNetworkError(error) {
    return error.name === 'TypeError' || error.message.includes('network')
  }
  
  async request(endpoint, options = {}, config = {}) {
    const { 
      retries = 2, 
      showError = true, 
      debounce = true, 
      timeout = 30000 
    } = config
    
    // 请求防抖
    if (debounce) {
      const key = this._generateKey(endpoint, options)
      if (this.pending.has(key)) {
        return this.pending.get(key)
      }
      const promise = this._doRequest(endpoint, options, retries, showError, timeout)
      this.pending.set(key, promise)
      return promise.finally(() => this.pending.delete(key))
    }
    
    return this._doRequest(endpoint, options, retries, showError, timeout)
  }
  
  async _doRequest(endpoint, options, retries, showError, timeout) {
    // 创建 AbortController 用于超时和取消
    const controller = new AbortController()
    const timeoutId = setTimeout(() => controller.abort(), timeout)
    this.abortControllers.set(endpoint, controller)
    
    try {
      const csrfToken = await getCSRFToken()
      const response = await fetch(`${this.baseUrl}${endpoint}`, {
        headers: { 
          'Content-Type': 'application/json',
          'X-CSRF-Token': csrfToken,
          ...options.headers 
        },
        signal: controller.signal,
        ...options
      })
      
      clearTimeout(timeoutId)
      
      if (!response.ok) {
        throw new Error(`HTTP ${response.status}: ${response.statusText}`)
      }
      
      return await response.json()
    } catch (error) {
      clearTimeout(timeoutId)
      
      // 超时错误
      if (error.name === 'AbortError') {
        const timeoutError = new Error('请求超时，请检查网络连接')
        if (showError) ElMessage.error(timeoutError.message)
        throw timeoutError
      }
      
      // 网络错误重试
      if (retries > 0 && this._isNetworkError(error)) {
        await this._delay(1000)
        return this._doRequest(endpoint, options, retries - 1, showError, timeout)
      }
      
      if (showError) {
        ElMessage.error(error.message || '网络请求失败')
      }
      throw error
    } finally {
      this.abortControllers.delete(endpoint)
    }
  }
  
  // 取消指定请求
  cancelRequest(endpoint) {
    const controller = this.abortControllers.get(endpoint)
    if (controller) {
      controller.abort()
      this.abortControllers.delete(endpoint)
    }
  }
  
  // 取消所有请求
  cancelAllRequests() {
    this.abortControllers.forEach(controller => controller.abort())
    this.abortControllers.clear()
  }
  
  // GET 请求
  async get(endpoint, params = {}, config = {}) {
    const searchParams = new URLSearchParams(
      Object.entries(params).filter(([, v]) => v !== undefined && v !== '')
    )
    const url = searchParams.toString() ? `${endpoint}?${searchParams}` : endpoint
    return this.request(url, { method: 'GET' }, config)
  }
  
  // POST 请求
  async post(endpoint, data = {}, config = {}) {
    return this.request(endpoint, {
      method: 'POST',
```

**方案**（支持前端分页和服务端分页）：
```javascript
// src/composables/usePagination.js
import { ref, computed } from 'vue'

/**
 * 分页 Hook
 * @param {number} defaultPageSize - 默认每页条数
 * @param {Ref<number>} total - 总条数（服务端分页时需要）
 */
export function usePagination(defaultPageSize = 10, total = ref(0)) {
  const currentPage = ref(1)
  const pageSize = ref(defaultPageSize)
  
  // 总页数
  const totalPages = computed(() => {
    return total.value ? Math.ceil(total.value / pageSize.value) : 0
  })
  
  // 前端分页：对数据进行切片
  const paginate = (data) => {
    if (!data || !Array.isArray(data)) return []
    const start = (currentPage.value - 1) * pageSize.value
    return data.slice(start, start + pageSize.value)
  }
  
  // 服务端分页：生成分页参数
  const getServerParams = () => ({
    page: currentPage.value,
    limit: pageSize.value
  })
  
  // 每页条数变化
  const handleSizeChange = (size) => {
    pageSize.value = size
    currentPage.value = 1 // 重置到第一页
  }
  
  // 页码变化
  const handleCurrentChange = (page) => {
    currentPage.value = page
  }
  
  // 跳转到指定页
  const jumpToPage = (page) => {
    if (page >= 1 && (totalPages.value === 0 || page <= totalPages.value)) {
      currentPage.value = page
      return true
    }
    return false
  }
  
  // 重置
  const reset = () => {
    currentPage.value = 1
    pageSize.value = defaultPageSize
  }
  
  return { 
    currentPage, 
    pageSize, 
    totalPages,
    paginate, 
    getServerParams,
    handleSizeChange, 
    handleCurrentChange,
    jumpToPage,
    reset
  }
}
```

```javascript
export function useUserFilters() {

    studentId.value = ''
    grade.value = ''
    classNum.value = ''
    subject.value = ''
    name.value = ''
  }

    if (studentId.value) params.student_id = studentId.value
    if (name.value) params.name = name.value

    if (classNum.value) params.class = classNum.value
    if (subject.value) params.subjectId = subject.value

  }

  })
  
  return { 
    studentId, 
    grade, 
    classNum, 
    subject,
    name,
    validate,
    reset, 
    toParams,
    hasActiveFilters
  }
```

### 5.5 旧代码处理策略

> ⚠️ **核心原则**：新功能稳定运行后再清理旧代码，不做激进重构。

#### 处理原则

| 原则 | 说明 |

### 10.3 P0-3 缓存策略详细影响分析

#### 风险识别
- `questionStore.js` 被前后台共享
- 前台页面调用 `questionStore.initialize()` 加载数据
- 缓存策略变更可能影响前台数据加载

#### 前台使用点
```javascript

await questionStore.initialize()

// SubcategoryView.vue:118
await questionStore.initialize()

// App.vue:32
await questionStore.loadData()
```

#### 应对方案（已实施）
采用**前后台缓存键分离**策略：

| 页面 | 缓存键 | 缓存时间 |
|------|--------|----------|
| 前台页面 | `coreData_v1` | 24 小时 |
| 后台页面 | `adminCoreData_v1` | 5 分钟 |

**优点**：
1. 前后台缓存完全隔离
2. 前台现有行为不变
3. 后台可独立刷新缓存
4. 缓存版本号支持强制刷新

### 10.4 前台测试清单

| 测试项 | 测试页面 | 预期结果 | 优先级 |
|--------|----------|----------|--------|
| 首次加载 | HomeView | 从服务器获取数据 | P0 |
| 刷新加载（缓存有效） | HomeView | 从缓存读取 | P0 |
| 刷新加载（缓存过期） | HomeView | 从服务器获取 | P0 |
| 答题页面数据加载 | QuizView | 正常加载题目 | P0 |
| 子类页面数据加载 | SubcategoryView | 正常加载分类 | P0 |
| 结果页面数据加载 | ResultView | 正常显示结果 | P1 |
| 前后台同时使用 | 多页面 | 缓存互不影响 | P0 |

### 10.5 前台兼容性保障措施

1. **缓存键隔离**：前后台使用不同缓存键，互不影响
2. **向后兼容**：保留旧 API 参数，逐步迁移
3. **渐进式重构**：新功能先在后台验证，前台可选使用
4. **充分测试**：每个阶段完成后测试前台页面
