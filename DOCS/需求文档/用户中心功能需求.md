## 一、背景与目标

### 1.1 现状问题

| 问题类型 | 具体表现 | 影响 |
|----------|----------|------|
| **缺少用户入口** | 用户无法集中查看个人信息和学习数据 | 体验不完整 |
| **错题分散** | 错题按学科分散，无法统一查看和练习 | 效率低下 |
| **进度不透明** | 学生不知道自己各知识点掌握情况 | 学习盲目 |
| **数据不直观** | 缺少可视化的学习报告和统计数据 | 缺少反馈 |

### 1.2 优化目标

| 指标 | 当前值 | 目标值 |
|------|--------|--------|
| 用户功能完整度 | 40% | > 90% |
| 数据可视化程度 | 20% | > 80% |
| 用户学习反馈 | 无 | 实时反馈 |

```javascript
// routes/error-book.js

/**
 * GET /api/error-book/user/:userId
 * 获取用户所有错题（支持筛选和分页）
 */
router.get('/user/:userId', async (req, res) => {
  const { userId } = req.params;
  const { subjectId, subcategoryId, sortBy, page = 1, limit = 20 } = req.query;
  
  // 构建查询
  let sql = `
    SELECT 
      ec.id,
      ec.question_id,
      ec.error_count,
      ec.correct_count,
      ec.created_at,
      q.content,
      q.type,
      q.difficulty,
      s.name as subject_name,
      sub.name as subcategory_name
    FROM error_collection ec
    INNER JOIN questions q ON ec.question_id = q.id
    INNER JOIN subjects s ON q.subject_id = s.id
    INNER JOIN subcategories sub ON q.subcategory_id = sub.id
    WHERE ec.user_id = ?
  `;
  
  const params = [userId];
  
  // 筛选条件
  if (subjectId) {
    sql += ' AND q.subject_id = ?';
    params.push(subjectId);
  }
  
  if (subcategoryId) {
    sql += ' AND q.subcategory_id = ?';
    params.push(subcategoryId);
  }
  
  // 排序
  const orderBy = {
    'latest': 'ec.created_at DESC',
    'oldest': 'ec.created_at ASC',
    'most_errors': 'ec.error_count DESC',
    'least_practiced': 'ec.correct_count ASC'
  }[sortBy] || 'ec.created_at DESC';
  
  sql += ` ORDER BY ${orderBy}`;
  
  // 分页
  const offset = (parseInt(page) - 1) * parseInt(limit);
  sql += ' LIMIT ? OFFSET ?';
  params.push(parseInt(limit), offset);
  
  const errors = await db.query(sql, params);
  
  // 获取总数
  let countSql = 'SELECT COUNT(*) as total FROM error_collection WHERE user_id = ?';
  const countParams = [userId];
  
  if (subjectId) {
    countSql += ' AND question_id IN (SELECT id FROM questions WHERE subject_id = ?)';
    countParams.push(subjectId);
  }
  
  const countResult = await db.get(countSql, countParams);
  
  // 学科统计
  const subjectStats = await db.query(`
    SELECT s.name, COUNT(*) as count
    FROM error_collection ec
    INNER JOIN questions q ON ec.question_id = q.id
    INNER JOIN subjects s ON q.subject_id = s.id
    WHERE ec.user_id = ?
    GROUP BY s.id
  `, [userId]);
  
  res.json({
    success: true,
    errors,
    total: countResult.total,
    subjectStats,
    page: parseInt(page),
    limit: parseInt(limit)
  });
});

/**
 * DELETE /api/error-book/:id
 * 移除错题
 */
router.delete('/:id', async (req, res) => {
  await db.run('DELETE FROM error_collection WHERE id = ?', [req.params.id]);
  res.json({ success: true, message: '错题已移除' });
});
```

```javascript
// routes/learning-progress.js

/**
 * GET /api/learning-progress/user/:userId
 * 获取用户学习进度
 */
router.get('/user/:userId', async (req, res) => {
  const { userId } = req.params;
  
  // 获取各知识点进度
  const progress = await db.query(`
    SELECT 
      lp.*,
      s.name as subject_name,
      sub.name as subcategory_name
    FROM learning_progress lp
    INNER JOIN subjects s ON lp.subject_id = s.id
    INNER JOIN subcategories sub ON lp.subcategory_id = sub.id
    WHERE lp.user_id = ?
    ORDER BY s.id, lp.mastery_level DESC
  `, [userId]);
  
  // 计算整体进度
  const overall = await db.get(`
    SELECT 
      COUNT(*) as total_subcategories,
      SUM(CASE WHEN mastery_level >= 80 THEN 1 ELSE 0 END) as mastered,
      SUM(CASE WHEN mastery_level >= 40 AND mastery_level < 80 THEN 1 ELSE 0 END) as learning,
      SUM(CASE WHEN mastery_level < 40 THEN 1 ELSE 0 END) as not_started,
      AVG(mastery_level) as avg_mastery
    FROM learning_progress
    WHERE user_id = ?
  `, [userId]);
  
  // 按学科分组统计
  const subjectProgress = await db.query(`
    SELECT 
      s.id,
      s.name as subject_name,
      COUNT(*) as total_subcategories,
      AVG(lp.mastery_level) as avg_mastery,
      SUM(CASE WHEN lp.mastery_level >= 80 THEN 1 ELSE 0 END) as mastered
    FROM learning_progress lp
    INNER JOIN subjects s ON lp.subject_id = s.id
    WHERE lp.user_id = ?
    GROUP BY s.id
    ORDER BY avg_mastery DESC
  `, [userId]);
  
  res.json({
    success: true,
    progress,
    overall: overall || {
      total_subcategories: 0,
      mastered: 0,
      learning: 0,
      not_started: 0,
      avg_mastery: 0
    },
    subjectProgress
  });
});
```

```javascript
// routes/answer-history.js

/**
 * GET /api/answer-history/user/:userId
 * 获取答题历史记录
 */
router.get('/user/:userId', async (req, res) => {
  const { userId } = req.params;
  const { startDate, endDate, subjectId, page = 1, limit = 20 } = req.query;
  
  let sql = `
    SELECT 
      ar.id,
      ar.total_questions,
      ar.correct_count,
      ar.created_at,
      s.name as subject_name,
      sub.name as subcategory_name
    FROM answer_records ar
    INNER JOIN subjects s ON ar.subject_id = s.id
    INNER JOIN subcategories sub ON ar.subcategory_id = sub.id
    WHERE ar.user_id = ?
  `;
  
  const params = [userId];
  
  if (startDate) {
    sql += ' AND ar.created_at >= ?';
    params.push(startDate);
  }
  
  if (endDate) {
    sql += ' AND ar.created_at <= ?';
    params.push(endDate);
  }
  
  if (subjectId) {
    sql += ' AND ar.subject_id = ?';
    params.push(subjectId);
  }
  
  const offset = (parseInt(page) - 1) * parseInt(limit);
  sql += ' ORDER BY ar.created_at DESC LIMIT ? OFFSET ?';
  params.push(parseInt(limit), offset);
  
  const records = await db.query(sql, params);
  
  // 获取总数
  let countSql = 'SELECT COUNT(*) as total FROM answer_records WHERE user_id = ?';
  const countParams = [userId];
  
  if (startDate) {
    countSql += ' AND created_at >= ?';
    countParams.push(startDate);
  }
  
  if (endDate) {
    countSql += ' AND created_at <= ?';
    countParams.push(endDate);
  }
  
  if (subjectId) {
    countSql += ' AND subject_id = ?';
    countParams.push(subjectId);
  }
  
  const countResult = await db.get(countSql, countParams);
  
  res.json({
    success: true,
    records,
    total: countResult.total,
    page: parseInt(page),
    limit: parseInt(limit)
  });
});

/**
 * GET /api/answer-history/stats/:userId
 * 获取答题统计
 */
router.get('/stats/:userId', async (req, res) => {
  const { userId } = req.params;
  
  // 时段分布
  const timeDistribution = await db.query(`
    SELECT 
      HOUR(created_at) as hour,
      COUNT(*) as count
    FROM answer_records
    WHERE user_id = ?
    GROUP BY HOUR(created_at)
    ORDER BY hour
  `, [userId]);
  
  // 正确率趋势（最近30天）
  const accuracyTrend = await db.query(`
    SELECT 
      DATE(created_at) as date,
      AVG(correct_count / total_questions * 100) as avg_accuracy
    FROM answer_records
    WHERE user_id = ?
    AND created_at >= DATE_SUB(NOW(), INTERVAL 30 DAY)
    GROUP BY DATE(created_at)
    ORDER BY date
  `, [userId]);
  
  // 学科能力
  const subjectAbility = await db.query(`
    SELECT 
      s.name as subject,
      AVG(ar.correct_count / ar.total_questions * 100) as avg_accuracy
    FROM answer_records ar
    INNER JOIN subjects s ON ar.subject_id = s.id
    WHERE ar.user_id = ?
    GROUP BY s.id
  `, [userId]);
  
  res.json({
    success: true,
    timeDistribution,
    accuracyTrend,
    subjectAbility
  });
});
```
