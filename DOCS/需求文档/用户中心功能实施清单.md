**路由**：`GET /api/users/:id/rank`

**功能**：获取用户的全校排名信息

**代码**：
```javascript
// 在 routes/users.js 文件末尾添加

// 获取用户排名
router.get('/:id/rank', async (req, res) => {
  try {
    const { id } = req.params;
    
    // 获取用户信息
    const user = await db.get('SELECT * FROM users WHERE id = ?', [id]);
    if (!user) {
      return res.status(404).json({ error: '用户不存在' });
    }
    
    // 计算排名（按积分）
    const rankQuery = `
      SELECT COUNT(*) + 1 as rank
      FROM users
      WHERE points > ?
    `;
    const rankResult = await db.get(rankQuery, [user.points]);
    
    // 获取总学生数
    const totalResult = await db.get('SELECT COUNT(*) as total FROM users');
    
    res.json({
      userId: user.id,
      rank: rankResult.rank,
      totalStudents: totalResult.total,
      points: user.points,
      percentile: Math.round((1 - rankResult.rank / totalResult.total) * 100)
    });
  } catch (error) {
    console.error('[getRank] 获取排名失败:', error);
    res.status(500).json({ error: '获取排名失败' });
  }
});
```

### API-2：错题统计 API（中优先级）

**文件**：`routes/error-collection.js`

**路由**：`GET /api/error-collection/stats/:userId`

**功能**：获取用户的错题统计信息

**代码**：
```javascript
// 在 routes/error-collection.js 文件末尾添加

// 获取错题统计
router.get('/stats/:userId', async (req, res) => {
  try {
    const { userId } = req.params;
    
    // 统计总错题数
    const totalQuery = `
      SELECT COUNT(DISTINCT question_id) as total
      FROM error_collection
      WHERE user_id = ?
    `;
    const totalResult = await db.get(totalQuery, [userId]);
    
    // 按学科分组统计
    const bySubjectQuery = `
      SELECT 
        q.subject_id,
        s.name as subject_name,
        COUNT(DISTINCT ec.question_id) as count
      FROM error_collection ec
      JOIN questions q ON ec.question_id = q.id
      JOIN subjects s ON q.subject_id = s.id
      WHERE ec.user_id = ?
      GROUP BY q.subject_id
    `;
    const bySubject = await db.all(bySubjectQuery, [userId]);
    
    // 统计已掌握（正确次数 >= 3）
    const masteredQuery = `
      SELECT COUNT(DISTINCT question_id) as mastered
      FROM error_collection
      WHERE user_id = ? AND correct_count >= 3
    `;
    const masteredResult = await db.get(masteredQuery, [userId]);
    
    res.json({
      total: totalResult.total || 0,
      bySubject: bySubject || [],
      mastered: masteredResult.mastered || 0,
      learning: (totalResult.total || 0) - (masteredResult.mastered || 0)
    });
  } catch (error) {
    console.error('[getStats] 获取错题统计失败:', error);
    res.status(500).json({ error: '获取错题统计失败' });
  }
});
```

```javascript
// routes/error-book.js

const express = require('express');
const router = express.Router();
const db = require('../services/database');

/**
 * GET /api/error-book/user/:userId
 * 获取用户所有错题
 */
router.get('/user/:userId', async (req, res) => {
  try {
    const { userId } = req.params;
    const { subjectId, subcategoryId, sortBy, page = 1, limit = 20 } = req.query;
    
    // 构建查询
    let sql = `
      SELECT 
        ec.id,
        ec.question_id,
        ec.error_count,
        ec.correct_count,
        ec.created_at,
        q.content,
        q.type,
        q.difficulty,
        s.name as subject_name,
        sub.name as subcategory_name
      FROM error_collection ec
      INNER JOIN questions q ON ec.question_id = q.id
      INNER JOIN subjects s ON q.subject_id = s.id
      INNER JOIN subcategories sub ON q.subcategory_id = sub.id
      WHERE ec.user_id = ?
    `;
    
    const params = [userId];
    
    // 筛选条件
    if (subjectId) {
      sql += ' AND q.subject_id = ?';
      params.push(subjectId);
    }
    
    if (subcategoryId) {
      sql += ' AND q.subcategory_id = ?';
      params.push(subcategoryId);
    }
    
    // 排序
    const orderBy = {
      'latest': 'ec.created_at DESC',
      'oldest': 'ec.created_at ASC',
      'most_errors': 'ec.error_count DESC',
      'least_practiced': 'ec.correct_count ASC'
    }[sortBy] || 'ec.created_at DESC';
    
    sql += ` ORDER BY ${orderBy}`;
    
    // 分页
    const offset = (parseInt(page) - 1) * parseInt(limit);
    sql += ' LIMIT ? OFFSET ?';
    params.push(parseInt(limit), offset);
    
    const errors = await db.query(sql, params);
    
    // 获取总数
    let countSql = 'SELECT COUNT(*) as total FROM error_collection WHERE user_id = ?';
    const countParams = [userId];
    
    if (subjectId) {
      countSql += ' AND question_id IN (SELECT id FROM questions WHERE subject_id = ?)';
      countParams.push(subjectId);
    }
    
    const countResult = await db.get(countSql, countParams);
    
    // 学科统计
    const subjectStats = await db.query(`
      SELECT s.name, COUNT(*) as count
      FROM error_collection ec
      INNER JOIN questions q ON ec.question_id = q.id
      INNER JOIN subjects s ON q.subject_id = s.id
      WHERE ec.user_id = ?
      GROUP BY s.id
    `, [userId]);
    
    res.json({
      success: true,
      errors,
      total: countResult.total,
      subjectStats,
      page: parseInt(page),
      limit: parseInt(limit)
    });
    
  } catch (error) {
    console.error('[获取错题列表] 失败:', error);
    res.status(500).json({ error: '获取错题列表失败' });
  }
});

/**
 * DELETE /api/error-book/:id
 * 移除错题
 */
router.delete('/:id', async (req, res) => {
  try {
    const { id } = req.params;
    
    await db.run('DELETE FROM error_collection WHERE id = ?', [id]);
    
    res.json({
      success: true,
      message: '错题已移除'
    });
    
  } catch (error) {
    console.error('[移除错题] 失败:', error);
    res.status(500).json({ error: '移除错题失败' });
  }
});

module.exports = router;
```

```javascript
// routes/learning-progress.js

const express = require('express');
const router = express.Router();
const db = require('../services/database');

/**
 * GET /api/learning-progress/user/:userId
 * 获取用户学习进度
 */
router.get('/user/:userId', async (req, res) => {
  try {
    const { userId } = req.params;
    
    // 获取各知识点进度
    const progress = await db.query(`
      SELECT 
        lp.*,
        s.name as subject_name,
        sub.name as subcategory_name
      FROM learning_progress lp
      INNER JOIN subjects s ON lp.subject_id = s.id
      INNER JOIN subcategories sub ON lp.subcategory_id = sub.id
      WHERE lp.user_id = ?
      ORDER BY s.id, lp.mastery_level DESC
    `, [userId]);
    
    // 计算整体进度
    const overall = await db.get(`
      SELECT 
        COUNT(*) as total_subcategories,
        SUM(CASE WHEN mastery_level >= 80 THEN 1 ELSE 0 END) as mastered,
        SUM(CASE WHEN mastery_level >= 40 AND mastery_level < 80 THEN 1 ELSE 0 END) as learning,
        SUM(CASE WHEN mastery_level < 40 THEN 1 ELSE 0 END) as not_started,
        AVG(mastery_level) as avg_mastery
      FROM learning_progress
      WHERE user_id = ?
    `, [userId]);
    
    // 按学科分组统计
    const subjectProgress = await db.query(`
      SELECT 
        s.id,
        s.name as subject_name,
        COUNT(*) as total_subcategories,
        AVG(lp.mastery_level) as avg_mastery,
        SUM(CASE WHEN lp.mastery_level >= 80 THEN 1 ELSE 0 END) as mastered
      FROM learning_progress lp
      INNER JOIN subjects s ON lp.subject_id = s.id
      WHERE lp.user_id = ?
      GROUP BY s.id
      ORDER BY avg_mastery DESC
    `, [userId]);
    
    res.json({
      success: true,
      progress,
      overall: overall || {
        total_subcategories: 0,
        mastered: 0,
        learning: 0,
        not_started: 0,
        avg_mastery: 0
      },
      subjectProgress
    });
    
  } catch (error) {
    console.error('[获取学习进度] 失败:', error);
    res.status(500).json({ error: '获取学习进度失败' });
  }
});

module.exports = router;
```

```javascript
// routes/answer-history.js

const express = require('express');
const router = express.Router();
const db = require('../services/database');

/**
 * GET /api/answer-history/user/:userId
 * 获取答题历史记录
 */
router.get('/user/:userId', async (req, res) => {
  try {
    const { userId } = req.params;
    const { startDate, endDate, subjectId, page = 1, limit = 20 } = req.query;
    
    let sql = `
      SELECT 
        ar.id,
        ar.total_questions,
        ar.correct_count,
        ar.created_at,
        s.name as subject_name,
        sub.name as subcategory_name
      FROM answer_records ar
      INNER JOIN subjects s ON ar.subject_id = s.id
      INNER JOIN subcategories sub ON ar.subcategory_id = sub.id
      WHERE ar.user_id = ?
    `;
    
    const params = [userId];
    
    if (startDate) {
      sql += ' AND ar.created_at >= ?';
      params.push(startDate);
    }
    
    if (endDate) {
      sql += ' AND ar.created_at <= ?';
      params.push(endDate);
    }
    
    if (subjectId) {
      sql += ' AND ar.subject_id = ?';
      params.push(subjectId);
    }
    
    const offset = (parseInt(page) - 1) * parseInt(limit);
    sql += ' ORDER BY ar.created_at DESC LIMIT ? OFFSET ?';
    params.push(parseInt(limit), offset);
    
    const records = await db.query(sql, params);
    
    res.json({ success: true, records });
    
  } catch (error) {
    console.error('[获取答题历史] 失败:', error);
    res.status(500).json({ error: '获取答题历史失败' });
  }
});

/**
 * GET /api/answer-history/stats/:userId
 * 获取答题统计
 */
router.get('/stats/:userId', async (req, res) => {
  try {
    const { userId } = req.params;
    
    // 时段分布
    const timeDistribution = await db.query(`
      SELECT 
        HOUR(created_at) as hour,
        COUNT(*) as count
      FROM answer_records
      WHERE user_id = ?
      GROUP BY HOUR(created_at)
      ORDER BY hour
    `, [userId]);
    
    // 正确率趋势（最近30天）
    const accuracyTrend = await db.query(`
      SELECT 
        DATE(created_at) as date,
        AVG(correct_count / total_questions * 100) as avg_accuracy
      FROM answer_records
      WHERE user_id = ?
      AND created_at >= DATE_SUB(NOW(), INTERVAL 30 DAY)
      GROUP BY DATE(created_at)
      ORDER BY date
    `, [userId]);
    
    res.json({ success: true, timeDistribution, accuracyTrend });
    
  } catch (error) {
    console.error('[获取答题统计] 失败:', error);
    res.status(500).json({ error: '获取答题统计失败' });
  }
});

module.exports = router;
```

### 🔴 MySQL 8.0 参数化查询（非常重要！）

**关键问题**：MySQL 8.0 + mysql2 驱动不支持 `VALUES ?` 批量插入语法！

详细说明见：**`DOCS/技术文档/MySQL8参数化查询注意事项.md`**

**正确示例**：
```javascript
// ✅ 正确：手动构建占位符
const items = [{ a: 1, b: 'x' }, { a: 2, b: 'y' }];

const placeholders = items.map(() => '(?, ?)').join(', ');
const values = items.flatMap(item => [item.a, item.b]);

await db.query(
  `INSERT INTO table (col1, col2) VALUES ${placeholders}`,
  values
);

// ❌ 错误：VALUES ? 语法不支持（会导致 SQL 语法错误）
await db.query(
  'INSERT INTO table (col1, col2) VALUES ?',
  [items.map(i => [i.a, i.b])]
);
```

**批量插入示例**：
```javascript
// 批量插入答题行为
const behaviors = [
  { userId: 1, questionId: 1, answerTime: 30, modifications: 0 },
  { userId: 1, questionId: 2, answerTime: 25, modifications: 1 }
];

// 1. 构建占位符：(?, ?, ?, ?), (?, ?, ?, ?)
const placeholders = behaviors.map(() => '(?, ?, ?, ?)').join(', ');

// 2. 扁平化参数：[1, 1, 30, 0, 1, 2, 25, 1]
const values = behaviors.flatMap(b => [
  b.userId,
  b.questionId,
  b.answerTime,
  b.modifications
]);

// 3. 执行 SQL
await db.query(
  `INSERT INTO answer_behavior 
   (user_id, question_id, answer_time, modifications) 
   VALUES ${placeholders}`,
  values
);
```

```javascript
router.get('/stats/:userId', async (req, res) => {

    const { userId } = req.params;

    const totalQuery = `
      SELECT COUNT(DISTINCT question_id) as total
      FROM error_collection
      WHERE user_id = ?
    `;
    const totalResult = await db.get(totalQuery, [userId]);
    
    // 按学科分组统计
    const bySubjectQuery = `
      SELECT 
        q.subject_id,
        s.name as subject_name,
        COUNT(DISTINCT ec.question_id) as count
      FROM error_collection ec
      JOIN questions q ON ec.question_id = q.id
      JOIN subjects s ON q.subject_id = s.id
      WHERE ec.user_id = ?
      GROUP BY q.subject_id
    `;
    const bySubject = await db.all(bySubjectQuery, [userId]);
    
    // 统计已掌握（正确次数 >= 3）
    const masteredQuery = `
      SELECT COUNT(DISTINCT question_id) as mastered
      FROM error_collection
      WHERE user_id = ? AND correct_count >= 3
    `;
    const masteredResult = await db.get(masteredQuery, [userId]);
    
    res.json({
      total: totalResult.total || 0,
      bySubject: bySubject || [],
      mastered: masteredResult.mastered || 0,
      learning: (totalResult.total || 0) - (masteredResult.mastered || 0)
    });
  } catch (error) {
    console.error('[getStats] 获取错题统计失败:', error);
```
