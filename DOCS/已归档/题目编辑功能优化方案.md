## 一、项目背景

### 1.1 当前问题

| 问题 | 现状 | 影响 |
|------|------|------|
| **编辑方式** | 弹窗编辑，每次需重新加载 | 效率低，不适合批量操作 |
| **选项编辑器** | 工具栏简单，不支持图片插入 | 无法在选项中添加图片 |
| **图片存储** | 粘贴图片以 Base64 存储在数据库 | 数据库膨胀、查询变慢、无法缓存 |
| **图片查看** | 用户答题页面图片无法放大 | 图片细节看不清，影响答题体验 |
| **听力题** | 仅支持基础音频上传 | 播放器功能简陋 |
| **批量操作** | 功能有限 | 无法批量编辑内容 |
| **草稿保存** | 无自动保存 | 意外关闭丢失数据 |

### 1.2 优化目标

1. **提升编辑效率**：分屏编辑模式，快速连续编辑
2. **增强富文本功能**：选项支持图片、上下标等
3. **优化图片存储**：粘贴图片自动上传，减少数据库负担
4. **改善图片查看**：用户答题页面支持图片放大预览
5. **改善听力题体验**：增强音频播放器
6. **兼容性保障**：用户页面同时支持 Base64 和 URL 图片显示
7. **安全可靠**：完善的文件上传安全检查和 XSS 防护

#### 3.1.1 图片上传接口

**接口定义**：


**后端实现** (`routes/upload.js`)：
```javascript
const express = require('express');
const router = express.Router();
const multer = require('multer');
const path = require('path');
const fs = require('fs');
const sharp = require('sharp'); // 图片处理库

// 配置存储
const storage = multer.diskStorage({
  destination: (req, file, cb) => {
    const uploadDir = path.join(__dirname, '../../images');
    
    // 确保目录存在
    if (!fs.existsSync(uploadDir)) {
      fs.mkdirSync(uploadDir, { recursive: true });
    }
    
    cb(null, uploadDir);
  },
  filename: (req, file, cb) => {
    // 生成唯一文件名
    const uniqueSuffix = Date.now() + '-' + Math.round(Math.random() * 1E9);
    const ext = path.extname(file.originalname).toLowerCase();
    cb(null, `image-${uniqueSuffix}${ext}`);
  }
});

// 文件过滤
const imageFileFilter = (req, file, cb) => {
  // 允许的图片类型
  const allowedTypes = ['image/jpeg', 'image/png', 'image/gif', 'image/webp'];
  
  if (!allowedTypes.includes(file.mimetype)) {
    return cb(new Error('不支持的图片格式，仅支持 JPG、PNG、GIF、WebP'), false);
  }
  
  // 检查文件扩展名
  const ext = path.extname(file.originalname).toLowerCase();
  const allowedExts = ['.jpg', '.jpeg', '.png', '.gif', '.webp'];
  
  if (!allowedExts.includes(ext)) {
    return cb(new Error('不支持的文件扩展名'), false);
  }
  
  cb(null, true);
};

// 配置 multer
const upload = multer({
  storage: storage,
  fileFilter: imageFileFilter,
  limits: {
    fileSize: 2 * 1024 * 1024 // 2MB
  }
});

// 图片上传路由
router.post('/image', upload.single('image'), async (req, res) => {
  try {
    if (!req.file) {
      return res.status(400).json({
        success: false,
        error: '未收到图片文件'
      });
    }
    
    // 图片安全检查
    const securityCheck = await validateImageSecurity(req.file.path);
    if (!securityCheck.safe) {
      // 删除不安全的文件
      fs.unlinkSync(req.file.path);
      return res.status(400).json({
        success: false,
        error: securityCheck.reason
      });
    }
    
    // 图片优化（压缩）
    const optimizedPath = await optimizeImage(req.file.path);
    
    // 返回访问 URL
    const imageUrl = `/images/${path.basename(optimizedPath)}`;
    
    res.json({
      success: true,
      url: imageUrl,
      filename: path.basename(optimizedPath),
      size: fs.statSync(optimizedPath).size
    });
    
  } catch (error) {
    console.error('图片上传失败:', error);
    
    // 清理可能的临时文件
    if (req.file && fs.existsSync(req.file.path)) {
      fs.unlinkSync(req.file.path);
    }
    
    res.status(500).json({
      success: false,
      error: '图片上传失败: ' + error.message
    });
  }
});

// 图片安全检查
async function validateImageSecurity(filePath) {
  try {
    const buffer = fs.readFileSync(filePath);
    
    // 1. 检查文件头（Magic Number）
    const magicNumbers = {
      'image/jpeg': [0xFF, 0xD8, 0xFF],
      'image/png': [0x89, 0x50, 0x4E, 0x47],
      'image/gif': [0x47, 0x49, 0x46],
      'image/webp': [0x52, 0x49, 0x46, 0x46]
    };
    
    // 获取文件类型
    const fileType = await detectFileType(buffer);
    if (!fileType) {
      return { safe: false, reason: '无法识别的文件类型' };
    }
    
    // 2. 检查是否包含恶意代码（基本的脚本检测）
    const dangerousPatterns = [
      /<script/i,
      /javascript:/i,
      /onerror\s*=/i,
      /onload\s*=/i,
      /onclick\s*=/i
    ];
    
    const content = buffer.toString('utf8', 0, Math.min(buffer.length, 1024));
    for (const pattern of dangerousPatterns) {
      if (pattern.test(content)) {
        return { safe: false, reason: '文件包含潜在恶意内容' };
      }
    }
    
    // 3. 检查图片尺寸（防止解析攻击）
    const metadata = await sharp(buffer).metadata();
    if (metadata.width > 10000 || metadata.height > 10000) {
      return { safe: false, reason: '图片尺寸过大' };
    }
    
    return { safe: true };
    
  } catch (error) {
    return { safe: false, reason: '文件安全检查失败: ' + error.message };
  }
}

// 图片优化
async function optimizeImage(filePath) {
  const ext = path.extname(filePath).toLowerCase();
  const optimizedPath = filePath.replace(ext, `-optimized${ext}`);
  
  await sharp(filePath)
    .resize(1920, 1920, { // 最大尺寸
      fit: 'inside',
      withoutEnlargement: true
    })
    .jpeg({ quality: 85 }) // JPEG 质量
    .png({ compressionLevel: 8 }) // PNG 压缩
    .toFile(optimizedPath);
  
  // 删除原始文件，使用优化后的文件
  fs.unlinkSync(filePath);
  fs.renameSync(optimizedPath, filePath);
  
  return filePath;
}

// 检测文件类型
async function detectFileType(buffer) {
  const signature = buffer.slice(0, 4).toString('hex');
  
  const signatures = {
    'ffd8ffe0': 'image/jpeg',
    'ffd8ffe1': 'image/jpeg',
    'ffd8ffdb': 'image/jpeg',
    '89504e47': 'image/png',
    '47494638': 'image/gif',
    '52494646': 'image/webp'
  };
  
  return signatures[signature.slice(0, 8)] || null;
}

module.exports = router;
```

```javascript
// 路径安全处理
function sanitizeFilePath(filename) {
  // 移除路径分隔符
  const safe = filename.replace(/[\/\\]/g, '');
  
  // 移除特殊字符
  const sanitized = safe.replace(/[^\w\-.]/g, '_');

// 验证文件路径
function validateFilePath(requestedPath, baseDir) {
  const resolved = path.resolve(baseDir, requestedPath);
  const normalized = path.normalize(resolved);
  
  // 确保路径在允许的目录内
  if (!normalized.startsWith(path.resolve(baseDir))) {
    throw new Error('非法的文件路径');
  }
  
  return normalized;
}
```

### 4.2 XSS 防护

#### 4.2.1 富文本内容过滤

```javascript
// utils/xss-filter.js

/**
 * XSS 过滤器
 * 用于清理富文本内容中的恶意代码
 */
class XSSFilter {
  constructor() {
    // 允许的 HTML 标签
    this.allowedTags = [
      'p', 'br', 'span', 'strong', 'em', 'u', 's', 'del',
      'h1', 'h2', 'h3', 'h4', 'h5', 'h6',
      'ul', 'ol', 'li',
      'blockquote', 'pre', 'code',
      'sub', 'sup',
      'img'
    ];
    
    // 允许的属性
    this.allowedAttributes = {
      'span': ['style', 'class'],
      'img': ['src', 'alt', 'style', 'class'],
      'p': ['style', 'class'],
      '*': ['class'] // 所有标签允许 class
    };
    
    // 允许的样式
    this.allowedStyles = [
      'color', 'background-color',
      'font-size', 'font-weight',
      'text-align', 'text-decoration'
    ];
  }
  
  /**
   * 清理 HTML 内容
   * @param {string} html - 原始 HTML
   * @returns {string} - 清理后的安全 HTML
   */
  sanitize(html) {
    if (!html) return '';
    
    // 1. 移除脚本标签
    let clean = html.replace(/<script\b[^<]*(?:(?!<\/script>)<[^<]*)*<\/script>/gi, '');
    
    // 2. 移除事件处理器
    clean = clean.replace(/\s*on\w+\s*=\s*["'][^"']*["']/gi, '');
    
    // 3. 移除 javascript: 协议
    clean = clean.replace(/javascript:/gi, '');
    
    // 4. 移除 data: 协议（除了图片）
    clean = clean.replace(/(?<!src=["'])data:(?!image\/)/gi, '');
    
    // 5. 清理 iframe, object, embed 等危险标签
    clean = clean.replace(/<(iframe|object|embed|form|input|button)[^>]*>/gi, '');
    clean = clean.replace(/<\/(iframe|object|embed|form|input|button)>/gi, '');
    
    // 6. 验证图片 src
    clean = this.validateImageSources(clean);
    
    return clean;
  }
  
  /**
   * 验证图片源
   */
  validateImageSources(html) {
    // 匹配图片标签
    return html.replace(/<img[^>]+src=["']([^"']+)["'][^>]*>/gi, (match, src) => {
      // 允许的图片源：相对路径、Base64 图片
      const isValidSrc = 
        src.startsWith('/images/') ||
        src.startsWith('/audio/') ||
        src.startsWith('data:image/');
      
      if (!isValidSrc) {
        return '<!-- 无效的图片源 -->';
      }
      
      return match;
    });
  }
  
  /**
   * 深度清理（更严格）
   * 用于存储到数据库前的最终清理
   */
  deepSanitize(html) {
    let clean = this.sanitize(html);
    
    // 移除所有 style 中的 url()
    clean = clean.replace(/url\s*\([^)]*\)/gi, '');
    
    // 移除 expression()
    clean = clean.replace(/expression\s*\([^)]*\)/gi, '');
    
    return clean;
  }
}

module.exports = new XSSFilter();
```

### 4.4 CSRF 防护

#### 4.4.1 为什么需要 CSRF 防护

| 攻击类型 | 风险 | 影响接口 |
|----------|------|----------|
| 伪造文件上传 | 恶意用户上传非法文件 | `/api/upload/image`, `/api/upload/audio` |
| 伪造题目修改 | 篡改题目内容 | `/api/questions/*` |
| 伪造批量操作 | 批量删除题目 | `/api/questions/batch/*` |

#### 4.4.2 实现方案

**方案选择**：Token 验证（适用于同源请求）

```javascript
// middleware/csrf.js
const crypto = require('crypto')

class CSRFProtection {
  constructor() {
    this.tokens = new Map() // token -> { expires, used }
    this.tokenExpiry = 2 * 60 * 60 * 1000 // 2小时有效
  }
  
  // 生成 CSRF Token
  generateToken(sessionId) {
    const token = crypto.randomBytes(32).toString('hex')
    this.tokens.set(token, {
      sessionId,
      expires: Date.now() + this.tokenExpiry,
      used: false
    })
    return token
  }
  
  // 验证 CSRF Token
  validateToken(token, sessionId) {
    const record = this.tokens.get(token)
    
    if (!record) {
      return { valid: false, reason: '无效的Token' }
    }
    
    if (Date.now() > record.expires) {
      this.tokens.delete(token)
      return { valid: false, reason: 'Token已过期' }
    }
    
    if (record.sessionId !== sessionId) {
      return { valid: false, reason: 'Token不匹配' }
    }
    
    // 单次使用后删除（可选）
    // this.tokens.delete(token)
    
    return { valid: true }
  }
  
  // 清理过期 Token
  cleanup() {
    const now = Date.now()
    for (const [token, record] of this.tokens) {
      if (now > record.expires) {
        this.tokens.delete(token)
      }
    }
  }
}

const csrf = new CSRFProtection()

// 定期清理过期 Token
setInterval(() => csrf.cleanup(), 60 * 60 * 1000) // 每小时清理

// CSRF Token 中间件
function csrfTokenMiddleware(req, res, next) {
  // 为需要 CSRF 保护的接口生成 Token
  if (req.session) {
    req.csrfToken = csrf.generateToken(req.session.id)
    res.locals.csrfToken = req.csrfToken
  }
  next()
}

// CSRF 验证中间件
function csrfVerifyMiddleware(req, res, next) {
  // GET 请求不需要验证
  if (req.method === 'GET') {
    return next()
  }
  
  // 从请求头或请求体获取 Token
  const token = req.headers['x-csrf-token'] || req.body._csrf
  
  if (!token) {
    return res.status(403).json({
      success: false,
      error: '缺少CSRF Token'
    })
  }
  
  const validation = csrf.validateToken(token, req.session?.id)
  
  if (!validation.valid) {
    return res.status(403).json({
      success: false,
      error: validation.reason
    })
  }
  
  next()
}

module.exports = { csrfTokenMiddleware, csrfVerifyMiddleware }
```

```javascript
/**
 * 上传队列，控制并发数
 * 支持进度显示、取消上传、状态管理
 */
export const uploadQueue = {
  queue: [],
  running: 0,
  maxConcurrent: 3,
  abortControllers: new Map(), // 用于取消上传
  listeners: new Set(), // 状态监听器
  
  // 添加状态监听
  subscribe(listener) {
    this.listeners.add(listener)
    return () => this.listeners.delete(listener)
  },
  
  // 通知状态变化
  notify() {
    const state = {
      pending: this.queue.length,
      running: this.running,
      total: this.queue.length + this.running
    }
    this.listeners.forEach(listener => listener(state))
  },
  
  // 添加上传任务
  async add(file, taskId = null) {
    const id = taskId || `${Date.now()}-${Math.random().toString(36).slice(2)}`
    const controller = new AbortController()
    this.abortControllers.set(id, controller)
    
    return new Promise((resolve, reject) => {
      this.queue.push({ file, resolve, reject, id, controller })
      this.notify()
      this.process()
    })
  },
  
  // 处理队列
  async process() {
    if (this.running >= this.maxConcurrent || this.queue.length === 0) {
      return
    }
    
    this.running++
    this.notify()
    const { file, resolve, reject, id, controller } = this.queue.shift()
    
    try {
      const url = await uploadImageWithRetry(file, 3, controller.signal)
      resolve(url)
    } catch (error) {
      if (error.name === 'AbortError') {
        reject(new Error('上传已取消'))
      } else {
        reject(error)
      }
    } finally {
      this.running--
      this.abortControllers.delete(id)
      this.notify()
      this.process()
    }
  },
  
  // 取消上传
  cancel(taskId) {
    const controller = this.abortControllers.get(taskId)
    if (controller) {
      controller.abort()
      this.abortControllers.delete(taskId)
    }
    
    // 从队列中移除
    const index = this.queue.findIndex(item => item.id === taskId)
    if (index !== -1) {
      this.queue.splice(index, 1)
      this.notify()
    }
  },
  
  // 取消所有上传
  cancelAll() {
    this.abortControllers.forEach(controller => controller.abort())
    this.abortControllers.clear()
    this.queue = []
    this.notify()
  },
  
  // 获取状态
  getState() {
    return {
      pending: this.queue.length,
      running: this.running,
      total: this.queue.length + this.running
    }
  }
}

/**
```
```javascript
 * 草稿存储工具
 * 支持多标签页隔离、版本迁移、大小限制

const DRAFT_VERSION = 2
const MAX_AGE = 24 * 60 * 60 * 1000 // 24小时
const MAX_SIZE = 4 * 1024 * 1024 // 4MB

class DraftStorage {
  constructor() {
    this.version = DRAFT_VERSION
    this.maxAge = MAX_AGE
    this.maxSize = MAX_SIZE
  }
  
  // 获取标签页唯一ID
  getTabId() {
    if (typeof sessionStorage === 'undefined') return 'default'
    
    if (!sessionStorage.getItem('tabId')) {
      sessionStorage.setItem('tabId', Date.now().toString(36) + Math.random().toString(36).substr(2))
    }
    return sessionStorage.getItem('tabId')
  }
  
  // 获取存储 key
  get key() {
    return `question_edit_draft_${this.getTabId()}`
  }
  
  // 保存草稿
  save(questionId, formData) {
    const draft = {
      version: this.version,
      id: questionId,
      isNew: !questionId,
      data: formData,
      timestamp: Date.now()
    }
    
    let jsonStr = JSON.stringify(draft)
    
    // 检查大小，必要时压缩
    if (jsonStr.length > this.maxSize) {
      draft.data = this.compressData(formData)
      jsonStr = JSON.stringify(draft)
      
      if (jsonStr.length > this.maxSize) {
        console.warn('草稿内容过大，无法保存')
        return false
      }
    }
    
    try {
      localStorage.setItem(this.key, jsonStr)
      return true
    } catch (e) {
      if (e.name === 'QuotaExceededError') {
        // 清理过期草稿后重试
        this.clearExpiredDrafts()
        try {
          localStorage.setItem(this.key, jsonStr)
          return true
        } catch (e2) {
          console.error('存储空间不足，草稿保存失败')
          return false
        }
      }
      throw e
    }
  }
  
  // 恢复草稿
  restore() {
    try {
      const draft = localStorage.getItem(this.key)
      if (!draft) return null
      
      const parsed = JSON.parse(draft)
      
      // 版本检查和迁移
      if (parsed.version !== this.version) {
        return this.migrate(parsed)
      }
      
      // 超时检查
      if (Date.now() - parsed.timestamp > this.maxAge) {
        this.clear()
        return null
      }
      
      return {
        id: parsed.id,
        isNew: parsed.isNew,
        data: parsed.data,
        timestamp: parsed.timestamp
      }
    } catch (e) {
      console.error('草稿解析失败:', e)
      this.clear()
      return null
    }
  }
  
  // 检查草稿状态
  hasDraft(currentQuestionId) {
    const draft = this.restore()
    if (!draft) return false
    
    // 新建题目检查
    if (!currentQuestionId && draft.isNew) return true
    
    // 编辑题目检查
    if (currentQuestionId && draft.id === currentQuestionId) return true
    
    // 有其他题目的草稿
    return {
      hasOtherDraft: true,
      draftId: draft.id,
      draftIsNew: draft.isNew,
      timestamp: draft.timestamp
    }
  }
  
  // 清除草稿
  clear() {
    localStorage.removeItem(this.key)
  }
  
  // 清理过期草稿
  clearExpiredDrafts() {
    const keys = Object.keys(localStorage)
    
    keys.forEach(key => {
      if (key.startsWith('question_edit_draft_')) {
        try {
          const draft = JSON.parse(localStorage.getItem(key))
          if (Date.now() - draft.timestamp > this.maxAge) {
            localStorage.removeItem(key)
          }
        } catch (e) {
          localStorage.removeItem(key)
        }
      }
    })
  }
  
  // 压缩数据（移除Base64图片）
  compressData(formData) {
    const compressed = { ...formData }
    
    if (compressed.content) {
      compressed.content = compressed.content.replace(
        /<img[^>]+src="data:image\/[^"]+"/g,
        '<img src="[图片数据已压缩，请重新上传]"'
      )
    }
    
    if (compressed.options) {
      compressed.options = compressed.options.map(opt =>
        opt?.replace(
          /<img[^>]+src="data:image\/[^"]+"/g,
          '<img src="[图片数据已压缩，请重新上传]"'
        ) || opt
      )
    }
    
    return compressed
  }
  
  // 版本迁移
  migrate(oldDraft) {
    const version = oldDraft.version || 1
    
    // v1 -> v2 迁移
    if (version === 1) {
      this.clear()
      return null
    }
    
    this.clear()
    return null
  }
}

export default new DraftStorage()
```

#### 6.1.4 IndexedDB 备选方案

**问题**：localStorage 有 5-10MB 限制，对于包含大量图片的富文本可能不够用

**解决方案**：优先使用 IndexedDB，localStorage 作为降级备选

```javascript
// src/utils/draftStorageIndexedDB.js

const DB_NAME = 'PSCG_Drafts'
const DB_VERSION = 1
const STORE_NAME = 'drafts'
const MAX_AGE = 24 * 60 * 60 * 1000 // 24小时

class IndexedDBDraftStorage {
  constructor() {
    this.db = null
    this.supported = 'indexedDB' in window
  }
  
  // 初始化数据库
  async init() {
    if (!this.supported) {
      console.warn('IndexedDB 不支持，使用 localStorage 降级')
      return false
    }
    
    return new Promise((resolve, reject) => {
      const request = indexedDB.open(DB_NAME, DB_VERSION)
      
      request.onerror = () => {
        console.error('IndexedDB 打开失败:', request.error)
        resolve(false)
      }
      
      request.onsuccess = () => {
        this.db = request.result
        resolve(true)
      }
      
      request.onupgradeneeded = (event) => {
        const db = event.target.result
        
        // 创建对象存储
        if (!db.objectStoreNames.contains(STORE_NAME)) {
          const store = db.createObjectStore(STORE_NAME, { keyPath: 'key' })
          store.createIndex('timestamp', 'timestamp', { unique: false })
        }
      }
    })
  }
  
  // 获取标签页 ID
  getTabId() {
    if (typeof sessionStorage === 'undefined') return 'default'
    
    if (!sessionStorage.getItem('tabId')) {
      sessionStorage.setItem('tabId', Date.now().toString(36) + Math.random().toString(36).substr(2))
    }
    return sessionStorage.getItem('tabId')
  }
  
  // 获取存储 key
  get key() {
    return `question_edit_draft_${this.getTabId()}`
  }
  
  // 保存草稿
  async save(questionId, formData) {
    const draft = {
      key: this.key,
      id: questionId,
      isNew: !questionId,
      data: formData,
      timestamp: Date.now()
    }
    
    // 尝试使用 IndexedDB
    if (this.db) {
      try {
        await this.saveToIndexedDB(draft)
        return true
      } catch (error) {
        console.warn('IndexedDB 保存失败，降级到 localStorage:', error)
      }
    }
    
    // 降级到 localStorage
    return this.saveToLocalStorage(draft)
  }
  
  // IndexedDB 保存
  saveToIndexedDB(draft) {
    return new Promise((resolve, reject) => {
      const transaction = this.db.transaction([STORE_NAME], 'readwrite')
      const store = transaction.objectStore(STORE_NAME)
      const request = store.put(draft)
      
      request.onsuccess = () => resolve(true)
      request.onerror = () => reject(request.error)
    })
  }
  
  // localStorage 保存（降级方案）
  saveToLocalStorage(draft) {
    try {
      let jsonStr = JSON.stringify(draft)
      
      // 检查大小（4MB 安全限制）
      if (jsonStr.length > 4 * 1024 * 1024) {
        // 压缩 Base64 图片
        draft.data = this.compressData(draft.data)
        jsonStr = JSON.stringify(draft)
        
        if (jsonStr.length > 4 * 1024 * 1024) {
          console.warn('草稿内容过大，无法保存')
          return false
        }
      }
      
      localStorage.setItem(this.key, jsonStr)
      return true
    } catch (e) {
      if (e.name === 'QuotaExceededError') {
        console.error('存储空间不足')
      }
      return false
    }
  }
  
  // 恢复草稿
  async restore() {
    // 尝试从 IndexedDB 获取
    if (this.db) {
      try {
        const draft = await this.getFromIndexedDB()
        if (draft && Date.now() - draft.timestamp < MAX_AGE) {
          return {
            id: draft.id,
            isNew: draft.isNew,
            data: draft.data,
            timestamp: draft.timestamp
          }
        }
      } catch (error) {
        console.warn('IndexedDB 读取失败，尝试 localStorage:', error)
      }
    }
    
    // 降级到 localStorage
    return this.getFromLocalStorage()
  }
  
  // IndexedDB 获取
  getFromIndexedDB() {
    return new Promise((resolve, reject) => {
      const transaction = this.db.transaction([STORE_NAME], 'readonly')
      const store = transaction.objectStore(STORE_NAME)
      const request = store.get(this.key)
      
      request.onsuccess = () => resolve(request.result)
      request.onerror = () => reject(request.error)
    })
  }
  
  // localStorage 获取
  getFromLocalStorage() {
    try {
      const draft = localStorage.getItem(this.key)
      if (!draft) return null
      
      const parsed = JSON.parse(draft)
      
      if (Date.now() - parsed.timestamp > MAX_AGE) {
        this.clear()
        return null
      }
      
      return {
        id: parsed.id,
        isNew: parsed.isNew,
        data: parsed.data,
        timestamp: parsed.timestamp
      }
    } catch (e) {
      return null
    }
  }
  
  // 清除草稿
  async clear() {
    if (this.db) {
      try {
        const transaction = this.db.transaction([STORE_NAME], 'readwrite')
        const store = transaction.objectStore(STORE_NAME)
        store.delete(this.key)
      } catch (error) {
        // 忽略错误
      }
    }
    
    localStorage.removeItem(this.key)
  }
  
  // 压缩数据
  compressData(formData) {
    const compressed = { ...formData }
    
    if (compressed.content) {
      compressed.content = compressed.content.replace(
        /<img[^>]+src="data:image\/[^"]+"/g,
        '<img src="[图片数据已压缩，请重新上传]"'
      )
    }
    
    if (compressed.options) {
      compressed.options = compressed.options.map(opt =>
        opt?.replace(
          /<img[^>]+src="data:image\/[^"]+"/g,
          '<img src="[图片数据已压缩，请重新上传]"'
        ) || opt
      )
    }
    
    return compressed
  }
  
  // 检查草稿状态
  async hasDraft(currentQuestionId) {
    const draft = await this.restore()
    if (!draft) return false
    
    if (!currentQuestionId && draft.isNew) return true
    if (currentQuestionId && draft.id === currentQuestionId) return true
    
    return {
      hasOtherDraft: true,
      draftId: draft.id,
      draftIsNew: draft.isNew,
      timestamp: draft.timestamp
    }
  }
}

// 导出单例
const storage = new IndexedDBDraftStorage()

// 初始化
storage.init().then(success => {
  if (!success) {
    console.log('使用 localStorage 作为草稿存储')
  }
})

export default storage
```

| 存储方式 | 容量限制 | 适用场景 |
|----------|----------|----------|
| localStorage | 5-10MB | 小型草稿，纯文本内容 |
| IndexedDB | 50MB-1GB（可申请更多） | 大型草稿，含图片内容 |
| 服务器端 | 无限制 | 需要跨设备同步 |

### 7.3 文件生命周期管理

#### 7.3.1 问题分析

| 场景 | 当前问题 | 影响 |
|------|----------|------|
| **删除草稿** | 图片/音频已上传，但草稿删除后文件不会被清理 | 垃圾文件积累 |
| **删除题目** | 关联的图片/音频不会被删除 | 垃圾文件积累 |
| **更换图片** | 旧图片不会被删除 | 存储空间浪费 |
| **上传后未保存** | 上传了图片但没保存题目就关闭 | 垃圾文件积累 |

#### 7.3.2 文件生命周期状态

```
文件状态流转：

1. 上传中 ──成功──→ 临时文件 (images/tmp-xxx.jpg)
         │
         └──失败──→ 无文件

2. 临时文件 ──保存题目──→ 正式文件 (images/image-xxx.jpg)
           │
           └──未保存──→ 24小时后清理

3. 正式文件 ──删除题目──→ 待清理 (标记删除)
           │
           └──更换图片──→ 待清理 (标记删除)

4. 待清理 ──定时任务──→ 删除
```

**方案三：定时清理脚本（简单实用）**

```javascript
// scripts/cleanup-files.js

const fs = require('fs')
const path = require('path')
const db = require('../services/database')

async function cleanupOrphanFiles() {
  console.log('开始清理孤立文件...')
  
  const imagesDir = path.join(__dirname, '../images')
  const audioDir = path.join(__dirname, '../audio')
  
  // 1. 获取数据库中所有引用的文件
  const [questions] = await db.query(`
    SELECT id, content, options, audio_url 
    FROM questions 
    WHERE content LIKE '%/images/%' 
       OR content LIKE '%/audio/%' 
       OR options LIKE '%/images/%' 
       OR audio_url IS NOT NULL
  `)
  
  const usedFiles = new Set()
  
  questions.forEach(q => {
    // 提取图片URL
    const imageMatches = [
      ...(q.content?.matchAll(/\/images\/[^"'\s]+/g) || []),
      ...(q.options?.matchAll(/\/images\/[^"'\s]+/g) || [])
    ]
    imageMatches.forEach(match => usedFiles.add(path.basename(match[0])))
    
    // 提取音频URL
    if (q.audio_url) {
      usedFiles.add(path.basename(q.audio_url))
    }
  })
  
  // 2. 清理未使用的图片
  let deletedImages = 0
  const imageFiles = fs.readdirSync(imagesDir)
  
  imageFiles.forEach(file => {
    // 跳过临时目录
    if (file === 'temp') return
    
    if (!usedFiles.has(file)) {
      const filePath = path.join(imagesDir, file)
      const stat = fs.statSync(filePath)
      const ageHours = (Date.now() - stat.mtime) / (1000 * 60 * 60)
      
      // 超过24小时的未引用文件
      if (ageHours > 24) {
        fs.unlinkSync(filePath)
        deletedImages++
        console.log(`删除孤立图片: ${file}`)
      }
    }
  })
  
  // 3. 清理未使用的音频
  let deletedAudio = 0
  const audioFiles = fs.readdirSync(audioDir)
  
  audioFiles.forEach(file => {
    if (file === 'temp') return
    
    if (!usedFiles.has(file)) {
      const filePath = path.join(audioDir, file)
      const stat = fs.statSync(filePath)
      const ageHours = (Date.now() - stat.mtime) / (1000 * 60 * 60)
      
      if (ageHours > 24) {
        fs.unlinkSync(filePath)
        deletedAudio++
        console.log(`删除孤立音频: ${file}`)
      }
    }
  })
  
  // 4. 清理临时目录中超过24小时的文件
  const tempDirs = [
    path.join(imagesDir, 'temp'),
    path.join(audioDir, 'temp')
  ]
  
  let deletedTemp = 0
  tempDirs.forEach(tempDir => {
    if (fs.existsSync(tempDir)) {
      const tempFiles = fs.readdirSync(tempDir)
      tempFiles.forEach(file => {
        const filePath = path.join(tempDir, file)
        const stat = fs.statSync(filePath)
        const ageHours = (Date.now() - stat.mtime) / (1000 * 60 * 60)
        
        if (ageHours > 24) {
          fs.unlinkSync(filePath)
          deletedTemp++
          console.log(`删除临时文件: ${file}`)
        }
      })
    }
  })
  
  console.log(`清理完成: 删除图片 ${deletedImages}, 音频 ${deletedAudio}, 临时文件 ${deletedTemp}`)
  
  return { deletedImages, deletedAudio, deletedTemp }
}

// 执行清理
cleanupOrphanFiles()
  .then(() => process.exit(0))
  .catch(err => {
    console.error('清理失败:', err)
    process.exit(1)
  })
```

```javascript
// 提取题目中的文件
function extractFilesFromQuestion(question) {
  const files = []
  
  // 提取图片
  const imageMatches = [
    ...(question.content?.matchAll(/\/images\/[^"'\s]+/g) || []),
    ...(question.options?.matchAll(/\/images\/[^"'\s]+/g) || [])
  ]
  imageMatches.forEach(match => files.push(match[0]))
  
  // 提取音频
  if (question.audio_url) {
    files.push(question.audio_url)
  }
  
  return files
}

// 检查文件是否被其他题目引用
async function checkFileReference(filePath) {
  const filename = path.basename(filePath)
  
  const [results] = await db.query(`
    SELECT COUNT(*) as count FROM questions 
    WHERE content LIKE ? 
       OR options LIKE ? 
       OR audio_url = ?
  `, [`%/images/${filename}%`, `%/images/${filename}%`, `/audio/${filename}`])
  
  return results[0].count > 0
}
```

```javascript
// src/utils/lazyLoad.js

/**
 * 图片懒加载指令
 * 使用 Intersection Observer API
 */
export const lazyLoadDirective = {
  mounted(el, binding) {
    const observer = new IntersectionObserver((entries) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          const img = entry.target
          img.src = binding.value
          img.classList.add('loaded')
          observer.unobserve(img)
        }
      })
    }, {
      rootMargin: '50px', // 提前 50px 开始加载
      threshold: 0.1
    })
    
    observer.observe(el)
    el._observer = observer
  },
  
  unmounted(el) {
    if (el._observer) {
      el._observer.disconnect()
    }
  }
}

/**
 * 富文本内容中的图片懒加载
 */
export function applyLazyLoadToContent(container) {
  const images = container.querySelectorAll('img[data-src]')
  
  const observer = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        const img = entry.target
        img.src = img.dataset.src
        img.removeAttribute('data-src')
        img.classList.add('loaded')
        observer.unobserve(img)
      }
    })
  }, {
    rootMargin: '100px',
    threshold: 0.1
  })
  
  images.forEach(img => observer.observe(img))
}

/**
 * 图片加载占位符样式
 */
const lazyLoadStyles = `
  img[data-src] {
    background: linear-gradient(90deg, #f0f0f0 25%, #e0e0e0 50%, #f0f0f0 75%);
    background-size: 200% 100%;
    animation: shimmer 1.5s infinite;
    min-height: 100px;
  }
  
  img.loaded {
    animation: fadeIn 0.3s ease-in;
  }
  
  @keyframes shimmer {
    0% { background-position: -200% 0; }

  }
```
