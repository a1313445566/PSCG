#### 任务 4.1：修改批量导入组件 - 解析逻辑

**文件**：`src/components/admin/question-management/BatchAddQuestion.vue`

**修改位置**：第 148-263 行（parseBatchQuestions 函数）

**修改内容**：添加判断题解析逻辑
```javascript
// 优先检查【判断】标记
if (line.includes('【判断】')) {
  if (currentQuestion) questions.push(currentQuestion)
  
  const contentMatch = line.match(/【判断】(.+?)[\(（]\s*[√×]\s*[\)）]/)
  const answerMatch = line.match(/[\(（]\s*([√×])\s*[\)）]/)
  
  if (contentMatch && answerMatch) {
    currentQuestion = {
      content: contentMatch[1].trim(),
      answer: answerMatch[1] === '√' ? 'A' : 'B',
      type: 'judgment',
      options: ['对', '错'],
      explanation: ''
    }
  }
  continue
}

// 检查判断题答案格式 (√) 或 (×)
const judgmentMatch = line.match(/^(?:(\d+[.、]\s*))?(.+?)[\(（]\s*([√×])\s*[\)）]/)
if (judgmentMatch) {
  if (currentQuestion) questions.push(currentQuestion)
  
  currentQuestion = {
    content: judgmentMatch[2].trim(),
    answer: judgmentMatch[3] === '√' ? 'A' : 'B',
    type: 'judgment',
    options: ['对', '错'],
    explanation: ''
  }
  continue
}
```
